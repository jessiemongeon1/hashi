4//! In-process `hashi-guardian` for integration tests. Two stages:
5//! [`GuardianHarness::start`] serves gRPC; [`GuardianHarness::finalize`]
6//! runs provisioner-init once hashi DKG output is on chain.
7
8use anyhow::Context;
9use anyhow::Result;
10use bitcoin::Network;
11use hashi_guardian::Enclave;
12use hashi_guardian::OperatorInitTestArgs;
13use hashi_guardian::create_operator_initialized_enclave;
14use hashi_guardian::rpc::GuardianGrpc;
15use hashi_types::committee::Committee as HashiCommittee;
16use hashi_types::guardian::BitcoinPubkey;
17use hashi_types::guardian::LimiterState;
18use hashi_types::guardian::WithdrawalConfig;
19use hashi_types::proto::guardian_service_server::GuardianServiceServer;
20use std::net::SocketAddr;
21use std::sync::Arc;
22use tokio::net::TcpListener;
23use tokio::sync::oneshot;
24use tokio::task::JoinHandle;
25use tonic::transport::Server;

27/// In-process guardian reachable over gRPC on a local TCP socket.
28pub struct GuardianHarness {
29    enclave: Arc<Enclave>,
30    endpoint: String,
31    shutdown_tx: Option<oneshot::Sender<()>>,
32    server_handle: Option<JoinHandle<()>>,
33}
34
35impl GuardianHarness {
36    /// Start an operator-init'd guardian. Withdrawal RPCs stay gated until
37    /// [`Self::finalize`] completes provisioner-init.
38    pub async fn start(network: Network) -> Result<Self> {
39        let enclave = create_operator_initialized_enclave(
40            OperatorInitTestArgs::default().with_network(network),
41        )
42        .await;
43
44        let listener = TcpListener::bind("127.0.0.1:0")
45            .await
46            .context("bind guardian harness listener")?;
47        let addr: SocketAddr = listener.local_addr()?;
48        let endpoint = format!("http://{addr}");
49
50        let (shutdown_tx, shutdown_rx) = oneshot::channel::<()>();
51        let svc = GuardianGrpc {
52            enclave: enclave.clone(),
53            setup_mode: false,
54        };
55        let incoming = tokio_stream::wrappers::TcpListenerStream::new(listener);
56        let server_handle = tokio::spawn(async move {
57            let result = Server::builder()
58                .add_service(GuardianServiceServer::new(svc))
59                .serve_with_incoming_shutdown(incoming, async move {
60                    let _ = shutdown_rx.await;
61                })
62                .await;
63            if let Err(e) = result {
64                tracing::warn!("guardian harness server exited: {e}");
65            }
66        });
67
68        Ok(Self {
69            enclave,
70            endpoint,
71            shutdown_tx: Some(shutdown_tx),
72            server_handle: Some(server_handle),
73        })
74    }
75
76    /// Complete provisioner-init using the committee and master pubkey from hashi DKG.
77    pub async fn finalize(
78        &self,
79        committee: HashiCommittee,
80        master_pubkey: BitcoinPubkey,
81        withdrawal_config: WithdrawalConfig,
82        limiter_state: LimiterState,
83    ) -> Result<()> {
84        hashi_guardian::test_utils::finalize_enclave(
85            &self.enclave,
86            committee,
87            master_pubkey,
88            withdrawal_config,
89            limiter_state,
90        )
91        .map_err(|e| anyhow::anyhow!("finalize guardian enclave: {e:?}"))?;
92
93        anyhow::ensure!(
94            self.enclave.is_fully_initialized(),
95            "guardian did not reach fully-initialized state"
96        );
97        Ok(())
98    }
99
100    pub fn endpoint(&self) -> &str {
101        &self.endpoint
102    }
103
104    pub fn enclave(&self) -> &Arc<Enclave> {
105        &self.enclave
106    }
107}
108
109impl Drop for GuardianHarness {
110    fn drop(&mut self) {
111        if let Some(tx) = self.shutdown_tx.take() {
112            let _ = tx.send(());
113        }
114        if let Some(handle) = self.server_handle.take() {
115            handle.abort();
116        }
117    }
118}

120pub fn default_test_withdrawal_config(committee: &HashiCommittee) -> WithdrawalConfig {
121    let total_weight = committee.total_weight();
122    let committee_threshold = total_weight.div_ceil(3) * 2;
123    WithdrawalConfig {
124        committee_threshold,
125        refill_rate_sats_per_sec: 0,
126        max_bucket_capacity_sats: 100_000_000,
127    }
128}
129
130pub fn full_bucket(config: &WithdrawalConfig) -> LimiterState {
131    LimiterState {
132        num_tokens_available: config.max_bucket_capacity_sats,
133        last_updated_at: 0,
134        next_seq: 0,
135    }
