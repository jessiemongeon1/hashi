32#[derive(Parser)]
33pub struct Args {
34    /// Comma-separated paths to validator DB backups.
35    #[arg(long, value_delimiter = ',')]
36    db_paths: Vec<PathBuf>,
37
38    /// Expected public key in hex (for verification).
39    #[arg(long)]
40    expected_pubkey: Option<String>,
41
42    /// Override the source epoch (the one whose KeyRotation output we want
43    /// to reconstruct). Defaults to the latest encryption-key epoch found
44    /// in the first DB.
45    #[arg(long)]
46    epoch: Option<u64>,
47}

49pub async fn run(args: Args, onchain_state: &OnchainState, chain_id: &str) -> anyhow::Result<()> {
50    if args.db_paths.is_empty() {
51        bail!("--db-paths is required");
52    }
53
54    let onchain_committee_epochs: Vec<u64> = {
55        let state = onchain_state.state();
56        state
57            .hashi()
58            .committees
59            .committees()
60            .keys()
61            .copied()
62            .collect()
63    };
64    println!("On-chain committee epochs: {onchain_committee_epochs:?}");
65    for db_path in &args.db_paths {
66        let db = Database::open(db_path).with_context(|| {
67            format!("failed to open DB for epoch listing: {}", db_path.display())
68        })?;
69        let latest = db
70            .latest_encryption_key_epoch()
71            .map_err(|e| anyhow!("failed to scan encryption keys: {e}"))?;
72        let mut present = Vec::new();
73        let mut rot_present = Vec::new();
74        let mut dkg_present = Vec::new();
75        if let Some(top) = latest {
76            for e in top.saturating_sub(15)..=top {
77                if db
78                    .get_encryption_key(e)
79                    .map_err(|err| anyhow!("get_encryption_key({e}): {err}"))?
80                    .is_some()
81                {
82                    present.push(e);
83                }
84                if !db
85                    .list_all_rotation_messages(e)
86                    .map_err(|err| anyhow!("list_all_rotation_messages({e}): {err}"))?
87                    .is_empty()
88                {
89                    rot_present.push(e);
90                }
91                if !db
92                    .list_all_dealer_messages(e)
93                    .map_err(|err| anyhow!("list_all_dealer_messages({e}): {err}"))?
94                    .is_empty()
95                {
96                    dkg_present.push(e);
97                }
98            }
99        }
100        println!(
101            "  {} enc_keys = {present:?} rot_msgs = {rot_present:?} dkg_msgs = {dkg_present:?} (latest enc = {latest:?})",
102            db_path.display()
103        );
104    }
105
106    let previous_epoch = match args.epoch {
107        Some(e) => {
108            println!("Using --epoch override: {e}");
109            e
110        }
111        None => detect_epoch(&args.db_paths[0])?,
112    };
113    let reconstruction_epoch = previous_epoch + 1;
114
115    // Build a recovery-local `CommitteeSet` that mirrors the on-chain
116    // committees but pins `committees[reconstruction_epoch]` to
117    // `previous_epoch`'s committee. `MpcManager::new` requires a committee at
118    // `reconstruction_epoch`, and reconstruction decrypts AVSS messages with
119    // `previous_epoch`'s encryption keys (per #502). Cloning instead of
120    // mutating `onchain_state` keeps the watcher free to rescrape without
121    // racing this tool.
122    let recovery_committee_set =
123        build_recovery_committee_set(onchain_state, previous_epoch, reconstruction_epoch)?;
124    let previous_committee = recovery_committee_set
125        .committees()
126        .get(&previous_epoch)
127        .expect("previous_epoch was just inserted by build_recovery_committee_set");
128
129    println!(
130        "Previous epoch {previous_epoch}: {} validators, fetching certificates...",
131        previous_committee.members().len()
132    );
133
134    let raw_certs = fetch_certificates(onchain_state, previous_epoch, None)
135        .await
136        .map_err(|e| anyhow!("failed to fetch certificates: {e}"))?;
137    let certificates: Vec<CertificateV1> = raw_certs.into_iter().map(|(_, cert)| cert).collect();
138    println!(
139        "Fetched {} certificates for epoch {previous_epoch}",
140        certificates.len()
141    );
142
143    if certificates.is_empty() {
144        bail!("No certificates found for epoch {previous_epoch}. Try a different epoch.");
145    }
146
147    // For each validator DB, reconstruct their key shares
148    let dummy_signing_key = Bls12381PrivateKey::generate(&mut rand::thread_rng());
149    let mut all_shares: Vec<Eval<S>> = Vec::new();
150    let mut recovered_pubkey: Option<G> = None;
151
152    for (i, db_path) in args.db_paths.iter().enumerate() {
153        println!("\n=== Validator {i}: {} ===", db_path.display());
154
155        let db = Arc::new(
156            Database::open(db_path)
157                .with_context(|| format!("failed to open DB: {}", db_path.display()))?,
158        );
159        let Some(encryption_key) = db
160            .get_encryption_key(previous_epoch)
161            .map_err(|e| anyhow!("failed to read encryption key: {e}"))?
162        else {
163            println!("  Skipping: no encryption key for epoch {previous_epoch}");
164            continue;
165        };
166
167        let my_enc_pk = EncryptionPublicKey::from_private_key(&encryption_key);
168        let Some(validator_address) =
169            find_validator_by_encryption_key(previous_committee, &my_enc_pk)
170        else {
171            println!(
172                "  Skipping: DB encryption key for epoch {previous_epoch} doesn't match any committee member"
173            );
174            continue;
175        };
176        println!("  Validator address: {validator_address}");
177
178        let store = EpochPublicMessagesStore::new(db.clone(), previous_epoch);
179        let session_id = SessionId::new(chain_id, reconstruction_epoch, &ProtocolType::KeyRotation);
180        // `encryption_key` decrypted `previous_epoch`'s AVSS messages, which is
181        // what reconstruction reads. Pass it as both `encryption_key` (unused
182        // here) and `previous_encryption_key`.
183        let mut manager = MpcManager::new(
184            validator_address,
185            &recovery_committee_set,
186            reconstruction_epoch,
187            session_id,
188            encryption_key.clone(),
189            Some(encryption_key),
190            dummy_signing_key.clone(),
191            Box::new(store),
192            chain_id,
193            None, // weight_divisor
194            0,    // batch_size_per_weight (unused for reconstruction)
195            None, // test_corrupt_shares_for
196        )
197        .map_err(|e| anyhow!("failed to create MpcManager: {e}"))?;
198
199        // Override previous_epoch to match the backed-up DB's epoch, since
200        // the on-chain epoch may have advanced past the backup.
201        manager.set_previous_epoch(previous_epoch);
202
203        // Pass an empty complaint cache: the recovery tool runs reconstruction
204        // once without retrying through complaint recovery, so there are no
205        // recovered outputs to reuse across attempts.
206        let outcome = manager
207            .reconstruct_previous_output(&certificates, &std::collections::HashMap::new())
208            .map_err(|e| anyhow!("reconstruction failed: {e}"))?;
209
210        match outcome {
211            ReconstructionOutcome::Success(output) => {
212                println!(
213                    "  Public key: {}",
214                    hex::encode(output.public_key.to_byte_array())
215                );
216                println!("  Shares: {}", output.key_shares.shares.len());
217                println!("  Threshold: {}", output.threshold);
218
219                if let Some(ref pk) = recovered_pubkey
220                    && *pk != output.public_key
221                {
222                    bail!("Public key mismatch between validators!");
223                }
224                recovered_pubkey = Some(output.public_key);
225                all_shares.extend(output.key_shares.shares.iter().cloned());
226            }
227            ReconstructionOutcome::NeedsDkgComplaintRecovery { dealer_address, .. } => {
228                println!(
229                    "  Warning: needs DKG complaint recovery for dealer {dealer_address}, skipping"
230                );
231            }
232            ReconstructionOutcome::NeedsRotationComplaintRecovery {
233                dealer_address,
234                share_index,
235                ..
236            } => {
237                println!(
238                    "  Warning: needs rotation complaint recovery for dealer {dealer_address} share {share_index}, skipping"
239                );
240            }
241        }
242    }
243
244    // Lagrange interpolation to recover full private key
245    let pubkey =
246        recovered_pubkey.ok_or_else(|| anyhow!("no shares recovered from any validator"))?;
247    println!("\n=== Lagrange Interpolation ===");
248    println!("Total shares collected: {}", all_shares.len());
249
250    let secret_key = lagrange_interpolate_at_zero(&all_shares)?;
251
252    // Verify
253    let recovered_pk = G::generator() * secret_key;
254    println!(
255        "Recovered public key:  {}",
256        hex::encode(recovered_pk.to_byte_array())
257    );
258    println!(
259        "Expected public key:   {}",
260        hex::encode(pubkey.to_byte_array())
261    );
262
263    if recovered_pk != pubkey {
264        bail!("VERIFICATION FAILED: recovered key does not match expected public key");
265    }
266    println!("\nVerification PASSED!");
267
268    if let Some(ref expected) = args.expected_pubkey {
269        let expected_bytes = hex::decode(expected)?;
270        let expected_pk = G::from_byte_array(
271            &expected_bytes
272                .try_into()
273                .map_err(|_| anyhow!("invalid pubkey length"))?,
274        )
275        .map_err(|e| anyhow!("invalid expected pubkey: {e}"))?;
276        if recovered_pk != expected_pk {
277            bail!("MISMATCH with --expected-pubkey!");
278        }
279        println!("Matches --expected-pubkey!");
280    }
281
282    println!(
283        "\nRecovered private key (hex): {}",
284        hex::encode(secret_key.to_byte_array())
285    );
286
287    Ok(())
288}

290/// Clone the on-chain `CommitteeSet`'s committees and pin
291/// `committees[reconstruction_epoch]` to `previous_epoch`'s committee.
292///
293/// `MpcManager::new` requires `committees[reconstruction_epoch]` to exist and
294/// to be the committee whose encryption keys decrypt the AVSS messages we're
295/// reconstructing — which, per #502, is `previous_epoch`'s committee. This is
296/// true regardless of whether the cluster has published its real next-epoch
297/// committee yet, so we unconditionally overwrite; a warning is printed if an
298/// existing on-chain entry differs.
299fn build_recovery_committee_set(
300    onchain_state: &OnchainState,
301    previous_epoch: u64,
302    reconstruction_epoch: u64,
303) -> anyhow::Result<CommitteeSet> {
304    let state = onchain_state.state();
305    let mut committees = state.hashi().committees.committees().clone();
306    let previous_committee = committees
307        .get(&previous_epoch)
308        .cloned()
309        .ok_or_else(|| anyhow!("no on-chain committee for epoch {previous_epoch}"))?;
310    if let Some(existing) = committees.get(&reconstruction_epoch)
311        && existing != &previous_committee
312    {
313        println!(
314            "Warning: on-chain committee[{reconstruction_epoch}] differs from \
315             committee[{previous_epoch}]; recovery uses committee[{previous_epoch}]"
316        );
317    }
318    committees.insert(reconstruction_epoch, previous_committee);
319    let mut set = CommitteeSet::new(Address::ZERO, Address::ZERO);
320    set.set_committees(committees);
321    Ok(set)
322}

324fn detect_epoch(db_path: &std::path::Path) -> anyhow::Result<u64> {
325    let db = Database::open(db_path).with_context(|| {
326        format!(
327            "failed to open DB for epoch detection: {}",
328            db_path.display()
329        )
330    })?;
331    let epoch = db
332        .latest_encryption_key_epoch()
333        .map_err(|e| anyhow!("failed to scan encryption keys: {e}"))?
334        .ok_or_else(|| anyhow!("no encryption keys found in DB: {}", db_path.display()))?;
335    println!("Auto-detected source epoch: {epoch}");
336    Ok(epoch)
337}

339fn find_validator_by_encryption_key(
340    committee: &Committee,
341    enc_pk: &EncryptionPublicKey,
342) -> Option<Address> {
343    committee.members().iter().find_map(|m| {
344        if m.encryption_public_key().as_element().to_byte_array()
345            == enc_pk.as_element().to_byte_array()
346        {
347            Some(m.validator_address())
348        } else {
349            None
350        }
351    })
352}
353
354fn lagrange_interpolate_at_zero(shares: &[Eval<S>]) -> anyhow::Result<S> {
355    if shares.is_empty() {
356        bail!("no shares to interpolate");
357    }
358    let indices: Vec<S> = shares
359        .iter()
360        .map(|s| S::from(s.index.get() as u128))
361        .collect();
362    let mut result = S::zero();
363    for (i, share) in shares.iter().enumerate() {
364        let xi = indices[i];
365        let one = S::generator();
366        let mut numerator = one;
367        let mut denominator = one;
368        for (j, xj) in indices.iter().enumerate() {
369            if i == j {
370                continue;
371            }
372            numerator *= -*xj;
373            denominator *= xi - *xj;
374        }
375        let inv = denominator
376            .inverse()
377            .map_err(|e| anyhow!("Lagrange denominator inversion failed: {e}"))?;
378        let lagrange_coeff = numerator * inv;
379        result += share.value * lagrange_coeff;
380    }
381    Ok(result)
