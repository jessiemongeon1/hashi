4use crate::Enclave;
5use hashi_types::guardian::GuardianError;
6use hashi_types::guardian::GuardianResult;
7use std::sync::Arc;
8use std::time::Duration;
9use std::time::Instant;
10
11/// Stateful heartbeat writer.
12pub struct HeartbeatWriter {
13    pub enclave: Arc<Enclave>,
14    pub max_failures_interval: Duration,
15    /// a local record to track how many heartbeat attempts failed
16    pub consecutive_failures: u32,
17    /// records the last success timestamp
18    pub last_success_at: Option<Instant>,
19    /// records the first failure timestamp
20    pub failure_window_start_at: Option<Instant>,
21    /// the next sequence number used in s3 logs
22    pub next_seq: u64,
23}
24
25impl HeartbeatWriter {
26    pub fn new(enclave: Arc<Enclave>, max_failures_interval: Duration) -> Self {
27        Self {
28            enclave,
29            max_failures_interval,
30            consecutive_failures: 0,
31            last_success_at: None,
32            failure_window_start_at: None,
33            next_seq: 0,
34        }
35    }
36
37    /// Attempt to send one heartbeat.
38    ///
39    /// - If operator init is not complete, this is a no-op.
40    /// - On success, resets failure state.
41    /// - On S3 error, retries are driven by `run`; errors after elapsed failure window.
42    pub async fn tick(&mut self, now: Instant) -> GuardianResult<()> {
43        if !self.enclave.is_operator_init_complete() {
44            return Ok(());
45        }
46
47        match self.enclave.log_heartbeat(self.next_seq).await {
48            Ok(()) => {
49                self.consecutive_failures = 0;
50                self.last_success_at = Some(now);
51                self.failure_window_start_at = None;
52                self.next_seq += 1;
53            }
54            Err(e) => {
55                // other errors shouldn't occur due to checks in operator_init_complete
56                debug_assert!(matches!(e, GuardianError::S3Error(_)));
57
58                self.consecutive_failures += 1;
59                if self.failure_window_start_at.is_none() {
60                    self.failure_window_start_at = Some(now);
61                }
62
63                let reference = self
64                    .last_success_at
65                    .or(self.failure_window_start_at)
66                    .unwrap_or(now);
67                if now.duration_since(reference) > self.max_failures_interval {
68                    return Err(GuardianError::InternalError(format!(
69                        "Heartbeat has no successful write for {:?}; last success {:?}, failure window start {:?}, consecutive failures {}, latest error {:?}",
70                        self.max_failures_interval,
71                        self.last_success_at,
72                        self.failure_window_start_at,
73                        self.consecutive_failures,
74                        e
75                    )));
76                }
77            }
78        }
79
80        Ok(())
81    }
82
83    /// Run the periodic heartbeat loop.
84    pub async fn run(mut self, interval: Duration, retry_interval: Duration) -> GuardianResult<()> {
85        let mut delay = Duration::ZERO;
86        loop {
87            tokio::time::sleep(delay).await;
88            self.tick(Instant::now()).await?;
89            delay = if self.consecutive_failures > 0 {
90                retry_interval
91            } else {
92                interval
93            };
94        }
95    }
96}

111    fn mk_s3_logger(client: Client) -> S3Logger {
112        S3Logger::from_client_for_tests(S3Config::mock_for_testing(), client)
113    }
114
115    async fn mk_operator_initialized_enclave(s3_logger: S3Logger) -> Arc<Enclave> {
116        Enclave::create_operator_initialized_with(
117            OperatorInitTestArgs::default().with_s3_logger(s3_logger),
118        )
119        .await
120    }

122    #[tokio::test]
123    async fn test_heartbeat_fails_after_max_failure_interval() {
124        // Mock S3 client that always fails put_object.
125        let put_fail = mock!(Client::put_object)
126            .match_requests(|req| req.bucket() == Some("test-bucket"))
127            .sequence()
128            .http_status(500, None)
129            .times(10)
130            .build();
131
132        // Disable retries so each heartbeat attempt makes exactly one put_object call.
133        let client = mock_client!(aws_sdk_s3, RuleMode::MatchAny, &[&put_fail], |b| b
134            .retry_config(
135                aws_sdk_s3::config::retry::RetryConfig::standard().with_max_attempts(1)
136            ));
137
138        let enclave = mk_operator_initialized_enclave(mk_s3_logger(client)).await;
139
140        let max_failure_interval = Duration::from_secs(20);
141        let mut writer = HeartbeatWriter::new(enclave, max_failure_interval);
142        let t0 = Instant::now();
143
144        assert!(writer.tick(t0).await.is_ok());
145        assert_eq!(writer.consecutive_failures, 1);
146        assert_eq!(writer.last_success_at, None);
147        assert_eq!(writer.failure_window_start_at, Some(t0));
148
149        assert!(writer.tick(t0 + Duration::from_secs(10)).await.is_ok());
150        assert_eq!(writer.consecutive_failures, 2);
151        assert_eq!(writer.last_success_at, None);
152
153        assert!(writer.tick(t0 + Duration::from_secs(21)).await.is_err());
154        assert_eq!(put_fail.num_calls(), 3usize);
155    }

157    #[tokio::test]
158    async fn test_heartbeat_resets_failure_state_on_success() {
159        // fail twice, succeed, then fail again
160        let put_flaky = mock!(Client::put_object)
161            .match_requests(|req| req.bucket() == Some("test-bucket"))
162            .sequence()
163            .http_status(500, None)
164            .times(2)
165            .output(|| PutObjectOutput::builder().build())
166            .http_status(500, None)
167            .build();
168
169        // Disable retries so each heartbeat attempt makes exactly one put_object call.
170        let client = mock_client!(aws_sdk_s3, RuleMode::Sequential, &[&put_flaky], |b| b
171            .retry_config(
172                aws_sdk_s3::config::retry::RetryConfig::standard().with_max_attempts(1)
173            ));
174
175        let enclave = mk_operator_initialized_enclave(mk_s3_logger(client)).await;
176
177        let max_failure_interval = Duration::from_secs(30);
178        let mut writer = HeartbeatWriter::new(enclave, max_failure_interval);
179        let t0 = Instant::now();
180
181        assert!(writer.tick(t0).await.is_ok());
182        assert_eq!(writer.consecutive_failures, 1);
183        assert_eq!(writer.last_success_at, None);
184        assert_eq!(writer.failure_window_start_at, Some(t0));
185
186        assert!(writer.tick(t0 + Duration::from_secs(10)).await.is_ok());
187        assert_eq!(writer.consecutive_failures, 2);
188        assert_eq!(writer.last_success_at, None);
189
190        assert!(writer.tick(t0 + Duration::from_secs(15)).await.is_ok());
191        assert_eq!(writer.consecutive_failures, 0, "should reset");
192        assert_eq!(writer.next_seq, 1, "success should advance sequence");
193        assert_eq!(writer.last_success_at, Some(t0 + Duration::from_secs(15)));
194        assert_eq!(writer.failure_window_start_at, None);
195
196        assert!(writer.tick(t0 + Duration::from_secs(40)).await.is_ok());
197        assert_eq!(writer.consecutive_failures, 1);
198        assert_eq!(writer.last_success_at, Some(t0 + Duration::from_secs(15)));
199        assert_eq!(
200            writer.failure_window_start_at,
201            Some(t0 + Duration::from_secs(40))
202        );
203
204        assert_eq!(put_flaky.num_calls(), 4usize);
205    }
