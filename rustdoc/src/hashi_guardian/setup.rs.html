16/// Set up a new BTC key. Flow:
17///     1. KPs send their encryption pub keys to the operator
18///     2. Operator calls setup_new_key (and optionally returns its response to all KPs)
19///     3. KPs fetch the setup_new_key response from S3
20pub async fn setup_new_key(
21    enclave: Arc<Enclave>,
22    request: SetupNewKeyRequest,
23) -> GuardianResult<GuardianSigned<SetupNewKeyResponse>> {
24    info!("/setup_new_key - Received request.");
25    if !enclave.is_operator_init_complete() {
26        return Err(InvalidInputs("call operator_init first".into()));
27    }
28
29    let key_provisioner_pks = request.public_keys();
30    info!("Received {} public keys.", key_provisioner_pks.len());
31
32    info!("Generating new Bitcoin private key.");
33    let sk = SecretKey::random(&mut rand::thread_rng());
34    info!(
35        "Bitcoin key generated with fingerprint {:x}.",
36        fingerprint(&sk)
37    );
38
39    info!(
40        "Splitting secret into {} shares (threshold: {}).",
41        NUM_OF_SHARES, THRESHOLD
42    );
43    let shares = split_secret(&sk, &mut rand::thread_rng());
44
45    info!("Encrypting shares for key provisioners.");
46    let mut encrypted_shares = vec![];
47    let mut share_commitments = vec![];
48    for i in 0..NUM_OF_SHARES {
49        let share = &shares[i];
50        let pk = &key_provisioner_pks[i];
51        let encrypted = encrypt_share(share, pk, None, &mut rand::thread_rng());
52        let commitment = commit_share(share);
53        encrypted_shares.push(encrypted);
54        share_commitments.push(commitment);
55    }
56    info!("All {} shares encrypted.", NUM_OF_SHARES);
57
58    // Log to S3. KPs check that S3 has exactly one SetupNewKeySuccess message,
59    // which ensures that KPs receive consistent shares w.r.t each other.
60    enclave
61        .log_init(SetupNewKeySuccess {
62            encrypted_shares: encrypted_shares.clone(),
63            share_commitments: share_commitments.clone(),
64        })
65        .await?;
66
67    let response = enclave.sign(SetupNewKeyResponse {
68        encrypted_shares,
69        share_commitments: ShareCommitments::new(share_commitments)?,
70    });
71
72    Ok(response)
73}

75#[cfg(test)]
76mod tests {
77    use super::*;
78    use hashi_types::guardian::commit_share;
79    use hashi_types::guardian::decrypt_share;
80    use hashi_types::guardian::NUM_OF_SHARES;
81    use hpke::kem::X25519HkdfSha256;
82    use hpke::Kem;
83
84    fn mock_setup_new_key_request() -> (SetupNewKeyRequest, Vec<EncSecKey>) {
85        let mut private_keys = vec![];
86        let mut public_keys = vec![];
87        for _i in 0..NUM_OF_SHARES {
88            let mut rng = rand::thread_rng();
89            let (sk, pk) = X25519HkdfSha256::gen_keypair(&mut rng);
90            private_keys.push(sk);
91            public_keys.push(pk);
92        }
93
94        (SetupNewKeyRequest::new(public_keys).unwrap(), private_keys)
95    }
96
97    #[tokio::test]
98    async fn test_setup_new_key() {
99        let enclave = Enclave::create_operator_initialized().await;
100        let verification_key = &enclave.signing_pubkey();
101        let (request, kp_private_keys) = mock_setup_new_key_request();
102        let resp = setup_new_key(enclave.clone(), request).await.unwrap();
103        let validated_resp = resp.verify(verification_key).unwrap();
104        assert_eq!(validated_resp.encrypted_shares.len(), NUM_OF_SHARES);
105
106        for (enc_share, sk) in validated_resp
107            .encrypted_shares
108            .iter()
109            .zip(kp_private_keys.iter())
110            .take(NUM_OF_SHARES)
111        {
112            let share = decrypt_share(enc_share, sk, None).unwrap();
113            let commitment = commit_share(&share);
114            assert!(validated_resp.share_commitments.contains(&commitment));
115            println!("Received share: (id) {:?}", enc_share.id);
116        }
117    }
