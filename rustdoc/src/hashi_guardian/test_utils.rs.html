16/// Mock S3 logger that returns success for every PutObject call.
17pub fn mock_logger() -> S3Logger {
18    use aws_sdk_s3::operation::put_object::PutObjectOutput;
19    use aws_sdk_s3::Client;
20    use aws_smithy_mocks::mock;
21    use aws_smithy_mocks::mock_client;
22    use aws_smithy_mocks::RuleMode;
23    use hashi_types::guardian::S3Config;
24
25    let put_ok = mock!(Client::put_object).then_output(|| PutObjectOutput::builder().build());
26    let client = mock_client!(aws_sdk_s3, RuleMode::MatchAny, &[&put_ok]);
27    S3Logger::from_client_for_tests(S3Config::mock_for_testing(), client)
28}

30pub struct OperatorInitTestArgs {
31    pub network: Network,
32    pub commitments: ShareCommitments,
33    pub s3_logger: S3Logger,
34}
35
36impl Default for OperatorInitTestArgs {
37    fn default() -> Self {
38        let commitments = (1..=NUM_OF_SHARES)
39            .map(|id| ShareCommitment {
40                id: std::num::NonZeroU16::new(id as u16).unwrap(),
41                digest: vec![],
42            })
43            .collect();
44
45        Self {
46            network: Network::Regtest,
47            commitments: ShareCommitments::new(commitments).unwrap(),
48            s3_logger: mock_logger(),
49        }
50    }
51}

53impl OperatorInitTestArgs {

57    }

59    pub fn with_commitments(mut self, commitments: ShareCommitments) -> Self {

70impl Enclave {
71    pub fn create_with_random_keys() -> Arc<Self> {
72        let signing_keys = GuardianSignKeyPair::new(rand::thread_rng());
73        let encryption_keys = GuardianEncKeyPair::random(&mut rand::thread_rng());
74        Arc::new(Enclave::new(signing_keys, encryption_keys))
75    }
76
77    /// Create an enclave post operator_init() but pre provisioner_init().
78    pub async fn create_operator_initialized() -> Arc<Self> {
79        Self::create_operator_initialized_with(OperatorInitTestArgs::default()).await
80    }
81
82    pub async fn create_operator_initialized_with(args: OperatorInitTestArgs) -> Arc<Self> {
83        let enclave = Self::create_with_random_keys();
84        enclave.config.set_s3_logger(args.s3_logger).unwrap();
85        enclave.config.set_bitcoin_network(args.network).unwrap();
86        enclave.set_share_commitments(args.commitments).unwrap();
87        enclave
88            .scratchpad
89            .operator_init_logging_complete
90            .set(())
91            .expect("operator_init_logging_complete should only be set once");
92
93        assert!(enclave.is_operator_init_complete() && !enclave.is_provisioner_init_complete());
94        enclave
95    }
96}
97
98pub async fn create_operator_initialized_enclave(args: OperatorInitTestArgs) -> Arc<Enclave> {
99    Enclave::create_operator_initialized_with(args).await
100}

102pub struct FullyInitializedArgs {
103    pub network: Network,
104    pub committee: HashiCommittee,
105    pub master_pubkey: BitcoinPubkey,
106    pub withdrawal_config: WithdrawalConfig,
107    pub limiter_state: LimiterState,
108}
109
110/// Drive an operator-initialized enclave to fully-initialized state without
111/// running the share-encryption round-trip. Generates a fresh BTC keypair.
112pub fn finalize_enclave(
113    enclave: &Arc<Enclave>,
114    committee: HashiCommittee,
115    master_pubkey: BitcoinPubkey,
116    withdrawal_config: WithdrawalConfig,
117    limiter_state: LimiterState,
118) -> GuardianResult<()> {
119    let secp = Secp256k1::new();
120    let mut sk_bytes = [0u8; 32];
121    rand::thread_rng().fill_bytes(&mut sk_bytes);
122    let enclave_btc_keypair = Keypair::from_secret_key(
123        &secp,
124        &SecretKey::from_slice(&sk_bytes).expect("random bytes form a valid secp256k1 key"),
125    );
126    enclave.config.set_btc_keypair(enclave_btc_keypair)?;
127    enclave.config.set_hashi_btc_pk(master_pubkey)?;
128    enclave.config.set_withdrawal_config(withdrawal_config)?;
129
130    let init_state =
131        ProvisionerInitState::new(committee, withdrawal_config, limiter_state, master_pubkey)?;
132    enclave.state.init(init_state)?;
133
134    enclave
135        .scratchpad
136        .provisioner_init_logging_complete
137        .set(())
138        .map_err(|_| {
139            GuardianError::InvalidInputs("provisioner_init_logging_complete already set".into())
140        })?;
141    Ok(())
142}
143
144/// Operator-init + finalize in one shot.
145pub async fn create_fully_initialized_enclave(args: FullyInitializedArgs) -> Arc<Enclave> {
146    let FullyInitializedArgs {
147        network,
148        committee,
149        master_pubkey,
150        withdrawal_config,
151        limiter_state,
152    } = args;
153
154    let enclave =
155        create_operator_initialized_enclave(OperatorInitTestArgs::default().with_network(network))
156            .await;
157
158    finalize_enclave(
159        &enclave,
160        committee,
161        master_pubkey,
162        withdrawal_config,
163        limiter_state,
164    )
165    .expect("finalize_enclave should succeed on a fresh enclave");
166
167    assert!(enclave.is_fully_initialized());
168    enclave
169}
