144    pub fn set_mpc_manager(&self, manager: mpc::MpcManager) {
145        match self.mpc_manager.get() {
146            Some(lock) => {
147                // RwLock::write only fails if poisoned (a thread panicked while holding the lock).
148                // Poisoning indicates a bug, so we propagate the panic rather than recover.
149                *lock.write().unwrap() = manager;
150            }
151            None => {
152                // First-time initialization (e.g. new committee member joining mid-rotation).
153                let _ = self.mpc_manager.set(Arc::new(RwLock::new(manager)));
154            }
155        }
156    }

244    pub fn prepare_encryption_key(&self, epoch: u64) -> anyhow::Result<EncryptionPublicKey> {
245        if let Some(existing) = self.db.get_encryption_key(epoch)? {
246            return Ok(EncryptionPublicKey::from_private_key(&existing));
247        }
248        let private_key = EncryptionPrivateKey::new(&mut rand::thread_rng());
249        let public_key = EncryptionPublicKey::from_private_key(&private_key);
250        self.db
251            .store_encryption_key(epoch, &private_key)
252            .map_err(|e| anyhow!("failed to store encryption key for epoch {epoch}: {e}"))?;
253        Ok(public_key)
254    }

256    pub(crate) fn prepare_next_epoch_keys(&self, epoch: u64) -> anyhow::Result<NextEpochKeys> {
257        let encryption_public_key = self.prepare_encryption_key(epoch)?;
258        let signing_private_key = self.prepare_signing_key(epoch)?;
259        Ok(NextEpochKeys {
260            encryption_public_key,
261            signing_private_key,
262        })
263    }

265    pub async fn prepare_and_register_keys(self: &Arc<Self>, epoch: u64) -> anyhow::Result<()> {
266        let keys = self.prepare_next_epoch_keys(epoch)?;
267        let mut executor = sui_tx_executor::SuiTxExecutor::from_hashi(self.clone())?;
268        executor
269            .execute_register_or_update_validator(
270                &self.config,
271                None,
272                Some(&keys.encryption_public_key),
273                Some(&keys.signing_private_key),
274            )
275            .await
276            .map(|_| ())
277    }

279    pub(crate) fn backup_after_epoch_change(&self, epoch: u64) -> anyhow::Result<Option<PathBuf>> {
280        let Some(config_path) = self.config_path.as_deref() else {
281            tracing::warn!(
282                epoch,
283                "Skipping automatic backup: server config path is not set"
284            );
285            return Ok(None);
286        };
287        let Some(recipient) = self.config.backup_age_pubkey.as_ref() else {
288            tracing::warn!(
289                epoch,
290                "Skipping automatic backup: backup_age_pubkey is not configured"
291            );
292            return Ok(None);
293        };
294
295        let output_path = crate::backup::save(
296            config_path,
297            &self.config,
298            self.db.as_ref(),
299            recipient,
300            self.config.backup_dir(),
301        )?;
302        tracing::info!(
303            epoch,
304            output = %output_path.display(),
305            "Automatic backup completed after epoch change",
306        );
307        Ok(Some(output_path))
308    }

310    fn find_encryption_key_for_committee(
311        &self,
312        committee: &hashi_types::committee::Committee,
313        validator_address: sui_sdk_types::Address,
314        epoch: u64,
315    ) -> anyhow::Result<EncryptionPrivateKey> {
316        let member = committee
317            .members()
318            .iter()
319            .find(|m| m.validator_address() == validator_address)
320            .ok_or_else(|| anyhow!("validator not in committee for epoch {epoch}"))?;
321        let pub_key = member.encryption_public_key();
322        self.db
323            .find_encryption_key_matching(pub_key)
324            .map_err(|e| anyhow!("DB error looking up encryption key for epoch {epoch}: {e}"))?
325            .ok_or_else(|| {
326                anyhow!(
327                    "no DB encryption key matches committee record for epoch {epoch}; \
328                     operator intervention needed (re-register, possibly DB recovery)"
329                )
330            })
331    }

333    pub fn prepare_signing_key(&self, epoch: u64) -> anyhow::Result<Bls12381PrivateKey> {
334        if let Some(existing) = self.db.get_signing_key(epoch)? {
335            return Ok(existing);
336        }
337        let private_key = Bls12381PrivateKey::generate(&mut rand::thread_rng());
338        self.db
339            .store_signing_key(epoch, &private_key)
340            .map_err(|e| anyhow!("failed to store signing key for epoch {epoch}: {e}"))?;
341        Ok(private_key)
342    }

367    pub(crate) async fn next_reconfig_epoch(&self) -> anyhow::Result<u64> {
368        use sui_rpc::proto::sui::rpc::v2::GetServiceInfoRequest;
369        let mut client = self.onchain_state().client();
370        let service_info = client
371            .ledger_client()
372            .get_service_info(GetServiceInfoRequest::default())
373            .await?
374            .into_inner();
375        let sui_epoch = service_info.epoch();
376        let hashi_epoch = self.onchain_state().epoch();
377        let is_genesis = hashi_epoch == 0 && self.onchain_state().current_committee().is_none();
378        Ok(if is_genesis || hashi_epoch < sui_epoch {
379            sui_epoch
380        } else {
381            sui_epoch + 1
382        })
383    }

385    fn resolve_previous_encryption_key(
386        &self,
387        committee_set: &onchain::types::CommitteeSet,
388        target_epoch: u64,
389        validator_address: sui_sdk_types::Address,
390    ) -> anyhow::Result<Option<EncryptionPrivateKey>> {
391        let previous_committee_info = committee_set.previous_committee_for_target(target_epoch);
392        if previous_committee_info.is_none() && target_epoch > 0 {
393            let sui_epoch = self
394                .onchain_state_opt()
395                .map(|s| s.latest_checkpoint_epoch());
396            tracing::info!(
397                target_epoch,
398                committee_set_epoch = committee_set.epoch(),
399                pending_epoch_change = ?committee_set.pending_epoch_change(),
400                sui_epoch = ?sui_epoch,
401                "create_mpc_manager: target_epoch>0 with no previous committee recorded; \
402                 previous_encryption_key=None (genesis bootstrap onto chain at sui_epoch>0)"
403            );
404        }
405        previous_committee_info
406            .map(|(prev_ep, prev_committee)| {
407                self.find_encryption_key_for_committee(prev_committee, validator_address, prev_ep)
408                    .map(Some)
409                    .or_else(|e| {
410                        if !prev_committee
411                            .members()
412                            .iter()
413                            .any(|m| m.validator_address() == validator_address)
414                        {
415                            Ok(None)
416                        } else {
417                            Err(e)
418                        }
419                    })
420            })
421            .transpose()
422            .map(|opt| opt.flatten())
423    }

425    pub fn create_mpc_manager(
426        &self,
427        epoch: u64,
428        protocol_type: mpc::types::ProtocolType,
429    ) -> anyhow::Result<mpc::MpcManager> {
430        let state = self.onchain_state().state();
431        let hashi = state.hashi();
432        let committee_set = &hashi.committees;
433        let session_id = mpc::SessionId::new(self.config.sui_chain_id(), epoch, &protocol_type);
434        let validator_address = self.config.validator_address()?;
435        let encryption_key = self.find_encryption_key_for_committee(
436            committee_set
437                .committees()
438                .get(&epoch)
439                .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?,
440            validator_address,
441            epoch,
442        )?;
443        let previous_encryption_key =
444            self.resolve_previous_encryption_key(committee_set, epoch, validator_address)?;
445        let signing_key = self.find_signing_key_for_committee(
446            committee_set
447                .committees()
448                .get(&epoch)
449                .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?,
450            validator_address,
451            epoch,
452        )?;
453        let store = Box::new(storage::EpochPublicMessagesStore::new(
454            self.db.clone(),
455            epoch,
456        ));
457        let address = self.config.validator_address()?;
458        let chain_id = self.config.sui_chain_id();
459        let batch_size_per_weight =
460            if let Some(override_val) = self.config.test_batch_size_per_weight {
461                assert_test_only_config(
462                    chain_id,
463                    self.config.bitcoin_chain_id(),
464                    "test_batch_size_per_weight",
465                );
466                override_val
467            } else {
468                BATCH_SIZE_PER_WEIGHT
469            };
470        if self.config.test_corrupt_shares_for.is_some() {
471            assert_test_only_config(
472                chain_id,
473                self.config.bitcoin_chain_id(),
474                "test_corrupt_shares_for",
475            );
476        }
477        Ok(mpc::MpcManager::new(
478            address,
479            committee_set,
480            epoch,
481            session_id,
482            encryption_key,
483            previous_encryption_key,
484            signing_key,
485            store,
486            chain_id,
487            self.config.test_weight_divisor,
488            batch_size_per_weight,
489            self.config.test_corrupt_shares_for,
490        )?)
491    }

493    /// Verify the Sui RPC endpoint is on the expected chain.
494    async fn verify_sui_chain_id(&self) -> anyhow::Result<()> {
495        use sui_rpc::proto::sui::rpc::v2::GetServiceInfoRequest;
496
497        let sui_rpc_url = self.config.sui_rpc.as_deref().unwrap();
498        let mut client = sui_rpc::Client::new(sui_rpc_url)?;
499
500        let service_info = client
501            .ledger_client()
502            .get_service_info(GetServiceInfoRequest::default())
503            .await?
504            .into_inner();
505
506        let rpc_chain_id = service_info.chain_id();
507
508        let expected = self.config.sui_chain_id();
509
510        anyhow::ensure!(
511            rpc_chain_id == expected,
512            "Sui chain ID mismatch: local config has {expected}, \
513             but RPC endpoint reports {rpc_chain_id}"
514        );
515
516        tracing::info!("Sui chain ID verified: {expected}");
517        Ok(())
518    }

520    /// Verify the local config's `bitcoin_chain_id` matches the value stored on-chain.
521    fn verify_bitcoin_chain_id(&self) -> anyhow::Result<()> {
522        use bitcoin::hashes::Hash as _;
523        use std::str::FromStr;
524
525        let onchain_chain_id = self
526            .onchain_state()
527            .state()
528            .hashi()
529            .config
530            .bitcoin_chain_id()
531            .ok_or_else(|| anyhow!("bitcoin_chain_id not found in on-chain config"))?;
532
533        let local_chain_id = self.config.bitcoin_chain_id();
534        let block_hash = btc_monitor::config::BlockHash::from_str(local_chain_id)?;
535        let local_addr = sui_sdk_types::Address::new(*block_hash.as_byte_array());
536
537        anyhow::ensure!(
538            local_addr == onchain_chain_id,
539            "bitcoin chain ID mismatch: local config has {local_chain_id}, \
540             but on-chain value is {onchain_chain_id}"
541        );
542
543        tracing::info!("Bitcoin chain ID verified: {local_chain_id}");
544        Ok(())
545    }

595    pub async fn start(self: Arc<Self>) -> anyhow::Result<Service> {
596        let screener = if let Some(endpoint) = self.config.screener_endpoint() {
597            match grpc::screener_client::ScreenerClient::new(endpoint) {
598                Ok(client) => {
599                    tracing::info!("Screener client configured for {}", client.endpoint());
600                    Some(client)
601                }
602                Err(e) => {
603                    tracing::warn!(
604                        "Failed to configure screener client for {}: {}",
605                        endpoint,
606                        e
607                    );
608                    None
609                }
610            }
611        } else {
612            tracing::warn!("No screener endpoint configured; AML screening will be skipped");
613            None
614        };
615
616        self.metrics
617            .screener_enabled
618            .set(if screener.is_some() { 1 } else { 0 });
619
620        self.screener_client
621            .set(screener)
622            .map_err(|_| anyhow!("Screener client already initialized"))?;
623
624        // Verify Sui RPC is on the expected chain before loading any state.
625        self.verify_sui_chain_id().await?;
626
627        // Initialize on-chain state first so we can read guardian config from it.
628        let onchain_service = self.initialize_onchain_state().await?;
629
630        let guardian_endpoint = {
631            let state = self.onchain_state().state();
632            state.hashi().config.guardian_url().map(|s| s.to_string())
633        }
634        .or_else(|| self.config.guardian_endpoint().map(|s| s.to_string()));
635
636        let guardian = if let Some(endpoint) = guardian_endpoint.as_deref() {
637            match grpc::guardian_client::GuardianClient::new(endpoint) {
638                Ok(client) => {
639                    tracing::info!("Guardian client configured for {}", client.endpoint());
640                    Some(client)
641                }
642                Err(e) => {
643                    tracing::warn!(
644                        "Failed to configure guardian client for {}: {}",
645                        endpoint,
646                        e
647                    );
648                    None
649                }
650            }
651        } else {
652            tracing::info!("No guardian endpoint configured; guardian integration disabled");
653            None
654        };
655
656        self.metrics
657            .guardian_enabled
658            .set(if guardian.is_some() { 1 } else { 0 });
659        self.guardian_client
660            .set(guardian)
661            .map_err(|_| anyhow!("Guardian client already initialized"))?;
662
663        // Verify the local bitcoin_chain_id matches the on-chain value.
664        self.verify_bitcoin_chain_id()?;
665
666        // Sweep any SUI in the configured account to AB to enable parallelization of txns
667        sui_tx_executor::sweep_to_address_balance(&mut self.onchain_state().client(), &self.config)
668            .await?;
669
670        let next_epoch_keys = match self.next_reconfig_epoch().await {
671            Ok(next_epoch) => self
672                .prepare_next_epoch_keys(next_epoch)
673                .inspect_err(|e| {
674                    tracing::warn!(
675                        "Failed to prepare encryption/signing keys for epoch {next_epoch}: {e}; \
676                         will retry before next start_reconfig"
677                    )
678                })
679                .ok(),
680            Err(e) => {
681                tracing::warn!("Failed to compute next reconfig epoch: {e}");
682                None
683            }
684        };
685
686        // Register validator (if not already registered) and update any stale metadata.
687        match sui_tx_executor::SuiTxExecutor::from_config(&self.config, self.onchain_state())?
688            .execute_register_or_update_validator(
689                &self.config,
690                None,
691                next_epoch_keys.as_ref().map(|k| &k.encryption_public_key),
692                next_epoch_keys.as_ref().map(|k| &k.signing_private_key),
693            )
694            .await
695        {
696            Ok(true) => tracing::info!("Validator registered/updated on-chain"),
697            Ok(false) => tracing::debug!("Validator metadata is already up-to-date"),
698            Err(e) => tracing::warn!("Failed to register/update validator metadata: {e}"),
699        }
700
701        if self.is_in_current_committee() {
702            tracing::info!("Node is in the current committee; MPC service will recover state");
703        } else if self.onchain_state().epoch() == 0
704            && self.onchain_state().current_committee().is_none()
705        {
706            tracing::info!("No initial committee yet; MPC service will handle genesis bootstrap");
707        } else {
708            tracing::info!(
709                "Node is not in the current committee; skipping initial DKG manager creation"
710            );
711        }
712
713        let (backup_service, backup_handle) = backup::BackupService::new(self.clone());
714        let (mpc_service, mpc_handle) = mpc::MpcService::new(self.clone(), backup_handle);
715        self.mpc_handle
716            .set(mpc_handle)
717            .expect("MpcHandle already set");
718
719        let btc_monitor_service = self.initialize_btc_monitor().map_err(|e| {
720            tracing::error!("Failed to initialize BtcMonitor: {e}");
721            e
722        })?;
723
724        // Start services
725        let (_http_addr, http_service) = grpc::HttpService::new(self.clone()).start().await;
726        let leader_service = leader::LeaderService::new(self.clone()).start();
727        let backup_service = backup_service.start();
728        let mpc_service = mpc_service.start();
729        let guardian_bootstrap_service = self.clone().start_guardian_bootstrap();
730
731        let service = Service::new()
732            .merge(onchain_service)
733            .merge(btc_monitor_service)
734            .merge(http_service)
735            .merge(leader_service)
736            .merge(backup_service)
737            .merge(mpc_service)
738            .merge(guardian_bootstrap_service);
739
740        Ok(service)
741    }

743    async fn try_seed_guardian_state(&self) -> bool {
744        let Some(client) = self.guardian_client() else {
745            return false;
746        };
747        self.metrics.guardian_bootstrap_attempts_total.inc();
748        let rpc_start = std::time::Instant::now();
749        let rpc_result = client.get_guardian_info().await;
750        let rpc_elapsed = rpc_start.elapsed().as_secs_f64();
751        let info_pb = match rpc_result {
752            Ok(info) => {
753                self.metrics.record_guardian_rpc(
754                    metrics::GUARDIAN_RPC_METHOD_GET_GUARDIAN_INFO,
755                    metrics::GUARDIAN_RPC_OUTCOME_OK,
756                    rpc_elapsed,
757                );
758                info
759            }
760            Err(e) => {
761                self.metrics.record_guardian_rpc(
762                    metrics::GUARDIAN_RPC_METHOD_GET_GUARDIAN_INFO,
763                    metrics::GUARDIAN_RPC_OUTCOME_UNAVAILABLE,
764                    rpc_elapsed,
765                );
766                self.metrics.record_guardian_bootstrap_outcome(
767                    metrics::GUARDIAN_BOOTSTRAP_OUTCOME_RPC_FAILURE,
768                );
769                tracing::warn!("guardian bootstrap: GetGuardianInfo failed: {e}");
770                return false;
771            }
772        };
773        let info = match hashi_types::guardian::GetGuardianInfoResponse::try_from(info_pb) {
774            Ok(info) => info,
775            Err(e) => {
776                self.metrics.record_guardian_bootstrap_outcome(
777                    metrics::GUARDIAN_BOOTSTRAP_OUTCOME_PARSE_FAILURE,
778                );
779                tracing::warn!("guardian bootstrap: parse failed: {e:?}");
780                return false;
781            }
782        };
783        let _ = self.guardian_signing_pubkey.set(Some(info.signing_pub_key));
784        let (Some(state), Some(config)) = (info.limiter_state, info.limiter_config) else {
785            self.metrics.record_guardian_bootstrap_outcome(
786                metrics::GUARDIAN_BOOTSTRAP_OUTCOME_NO_LIMITER_YET,
787            );
788            tracing::debug!("guardian bootstrap: guardian has no limiter yet");
789            return false;
790        };
791        let limiter = Arc::new(guardian_limiter::LocalLimiter::new(config, state));
792        if self.local_limiter.set(limiter.clone()).is_ok() {
793            tracing::info!(
794                ?state,
795                ?config,
796                "Local guardian limiter seeded from GetGuardianInfo",
797            );
798            self.metrics.record_limiter_state(&state, &config);
799            self.metrics.guardian_limiter_initialized.set(1);
800            // Hand the same Arc to OnchainState so the watcher can advance
801            // it inline when WithdrawalSignedEvent fires.
802            self.onchain_state().set_local_limiter(limiter);
803        }
804        self.metrics
805            .record_guardian_bootstrap_outcome(metrics::GUARDIAN_BOOTSTRAP_OUTCOME_SUCCESS);
806        true
807    }
808
809    fn start_guardian_bootstrap(self: Arc<Self>) -> Service {
810        use backon::Retryable;
811        Service::new().spawn_aborting(async move {
812            if self.guardian_client().is_none() {
813                return Ok(());
814            }
815            let policy = backon::ExponentialBuilder::default()
816                .with_min_delay(std::time::Duration::from_secs(1))
817                .with_max_delay(std::time::Duration::from_secs(30))
818                .without_max_times();
819            let _ = (|| async {
820                if self.try_seed_guardian_state().await {
821                    Ok::<(), ()>(())
822                } else {
823                    Err(())
824                }
825            })
826            .retry(policy)
827            .await;
828            tracing::info!("Guardian bootstrap complete");
829            Ok(())
830        })
831    }

833    pub(crate) fn is_in_current_committee(&self) -> bool {
834        let address = match self.config.validator_address() {
835            Ok(a) => a,
836            Err(_) => return false,
837        };
838        self.onchain_state()
839            .current_committee()
840            .is_some_and(|c| c.index_of(&address).is_some())
841    }
