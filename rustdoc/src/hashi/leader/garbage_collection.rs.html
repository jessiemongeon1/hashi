22// const MAX_DEPOSIT_REQUEST_AGE_MS: u64 = 1000 * 60 * 60 * 24 * 3; // 3 days
23// const DEPOSIT_REQUEST_DELETE_DELAY_MS: u64 = 1000 * 60 * 60 * 24; // 1 day
24const MAX_DEPOSIT_REQUEST_AGE_MS: u64 = 1000 * 60 * 60 * 24; // 1 days
25const DEPOSIT_REQUEST_DELETE_DELAY_MS: u64 = 1000 * 60; // 1 minute
26const MAX_DEPOSIT_REQUEST_DELETIONS_PER_GC: usize = 500;
27
28const MAX_PROPOSAL_AGE_MS: u64 = 1000 * 60 * 60 * 24 * 7; // 7 days
29const PROPOSAL_DELETE_DELAY_MS: u64 = 1000 * 60 * 60 * 24; // 1 day
30
31impl LeaderService {
32    /// Check for and delete expired deposit requests.
33    /// Deposit requests are sorted by timestamp and deleted if they are older than
34    /// MAX_DEPOSIT_REQUEST_AGE_MS.
35    pub(crate) fn check_delete_expired_deposit_requests(&mut self, checkpoint_timestamp_ms: u64) {
36        if self.deposit_gc_task.is_some() {
37            debug!("Deposit GC task already in-flight, skipping");
38            return;
39        }
40
41        let mut deposit_requests = self.inner.onchain_state().deposit_requests();
42        deposit_requests.sort_by_key(|r| r.timestamp_ms);
43
44        let Some(oldest_request) = deposit_requests.first() else {
45            return;
46        };
47
48        if checkpoint_timestamp_ms
49            < oldest_request.timestamp_ms
50                + MAX_DEPOSIT_REQUEST_AGE_MS
51                + DEPOSIT_REQUEST_DELETE_DELAY_MS
52        {
53            return;
54        }
55
56        let expired_requests: Vec<_> = deposit_requests
57            .iter()
58            .filter(|r| checkpoint_timestamp_ms > r.timestamp_ms + MAX_DEPOSIT_REQUEST_AGE_MS)
59            .take(MAX_DEPOSIT_REQUEST_DELETIONS_PER_GC)
60            .cloned()
61            .collect();
62        if expired_requests.is_empty() {
63            return;
64        }
65
66        info!(
67            "Scheduling deletion of {} expired deposit requests",
68            expired_requests.len()
69        );
70
71        let inner = self.inner.clone();
72        self.deposit_gc_task = Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
73            Self::delete_expired_deposit_requests(inner, expired_requests).await
74        })));
75    }
76
77    async fn delete_expired_deposit_requests(
78        inner: Arc<crate::Hashi>,
79        expired_requests: Vec<DepositRequest>,
80    ) -> anyhow::Result<()> {
81        let count = expired_requests.len();
82        let mut executor = SuiTxExecutor::from_hashi(inner)?;
83        executor
84            .execute_delete_expired_deposit_requests(&expired_requests)
85            .await?;
86        info!("Successfully deleted {count} expired deposit requests");
87        Ok(())
88    }

90    /// Check for and delete expired proposals.
91    /// Proposals are sorted by timestamp and deleted if they are older than MAX_PROPOSAL_AGE_MS.
92    pub(crate) fn check_delete_proposals(&mut self, checkpoint_timestamp_ms: u64) {
93        debug!("Entering check_delete_proposals");
94
95        if self.proposal_gc_task.is_some() {
96            debug!("Proposal GC task already in-flight, skipping");
97            return;
98        }
99
100        let mut proposals = self.inner.onchain_state().proposals();
101        // Sort proposals by timestamp, from earliest to latest
102        proposals.sort_by_key(|p| p.timestamp_ms);
103
104        // Check if it's time to delete
105        let Some(oldest_proposal) = proposals.first() else {
106            return;
107        };
108
109        // If there aren't any proposals at least 8 days old (7 days expiry + 1 day delay), don't do anything
110        if checkpoint_timestamp_ms
111            < oldest_proposal.timestamp_ms + MAX_PROPOSAL_AGE_MS + PROPOSAL_DELETE_DELAY_MS
112        {
113            return;
114        }
115
116        // Find all expired proposals (older than 7 days)
117        let expired_proposals: Vec<_> = proposals
118            .iter()
119            .filter(|p| checkpoint_timestamp_ms > p.timestamp_ms + MAX_PROPOSAL_AGE_MS)
120            .cloned()
121            .collect();
122
123        if expired_proposals.is_empty() {
124            return;
125        }
126
127        info!(
128            "Scheduling deletion of {} expired proposals",
129            expired_proposals.len()
130        );
131
132        let inner = self.inner.clone();
133        self.proposal_gc_task = Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
134            Self::delete_expired_proposals(inner, expired_proposals).await
135        })));
136    }

138    /// If there are pending UTXO cleanups and no task in-flight, spawn a
139    /// background task to process the next one. When the queue is empty,
140    /// scans on-chain state for orphaned locked UTXOs whose withdrawal has
141    /// already been confirmed (handles the crash-between-confirm-and-cleanup
142    /// case).
143    pub(crate) fn check_cleanup_spent_utxos(&mut self) {
144        if self.utxo_cleanup_gc_task.is_some() {
145            debug!("UTXO cleanup GC task already in-flight, skipping");
146            return;
147        }
148
149        if self.pending_utxo_cleanups.is_empty() && self.utxo_cleanup_scan_needed {
150            self.discover_orphaned_utxo_cleanups();
151            if self.pending_utxo_cleanups.is_empty() {
152                self.utxo_cleanup_scan_needed = false;
153            }
154        }
155
156        let Some(cleanup) = self.pending_utxo_cleanups.pop_front() else {
157            return;
158        };
159
160        info!(
161            utxo_count = cleanup.utxo_ids.len(),
162            "Scheduling UTXO cleanup for spent UTXOs",
163        );
164
165        let inner = self.inner.clone();
166        self.utxo_cleanup_gc_task = Some(AbortOnDropHandle::new(tokio::task::spawn(async move {
167            Self::cleanup_spent_utxos(inner, cleanup).await
168        })));
169    }
170
171    /// Scan local state for UTXOs with `spent_epoch` set — they have been
172    /// spent but their cleanup never ran (e.g. leader crashed).
173    fn discover_orphaned_utxo_cleanups(&mut self) {
174        let state = self.inner.onchain_state().state();
175        let utxo_records = state.hashi().utxo_pool.utxo_records();
176
177        let orphaned_ids = find_spent_utxos_pending_cleanup(utxo_records);
178
179        if !orphaned_ids.is_empty() {
180            info!(
181                "Discovered {} spent UTXO(s) pending cleanup",
182                orphaned_ids.len(),
183            );
184            self.pending_utxo_cleanups.push_back(PendingUtxoCleanup {
185                utxo_ids: orphaned_ids,
186            });
187        }
188    }
189
190    async fn cleanup_spent_utxos(
191        inner: Arc<crate::Hashi>,
192        cleanup: PendingUtxoCleanup,
193    ) -> anyhow::Result<()> {
194        let mut executor = SuiTxExecutor::from_hashi(inner)?;
195        executor
196            .execute_cleanup_spent_utxos(&cleanup.utxo_ids)
197            .await?;
198        info!(
199            utxo_count = cleanup.utxo_ids.len(),
200            "Successfully cleaned up spent UTXOs",
201        );
202        Ok(())
203    }

205    async fn delete_expired_proposals(
206        inner: Arc<crate::Hashi>,
207        expired_proposals: Vec<Proposal>,
208    ) -> anyhow::Result<()> {
209        use sui_sdk_types::Identifier;
210        use sui_sdk_types::StructTag;
211        use sui_sdk_types::TypeTag;
212        use sui_transaction_builder::Function;
213        use sui_transaction_builder::ObjectInput;
214        use sui_transaction_builder::TransactionBuilder;
215
216        let mut executor = SuiTxExecutor::from_hashi(inner.clone())?;
217        let hashi_ids = inner.config.hashi_ids();
218
219        let mut builder = TransactionBuilder::new();
220
221        let hashi_arg = builder.object(
222            ObjectInput::new(hashi_ids.hashi_object_id)
223                .as_shared()
224                .with_mutable(true),
225        );
226
227        // Clock object (0x6) - immutable shared object
228        let clock_arg = builder.object(
229            ObjectInput::new(Address::from_static("0x6"))
230                .as_shared()
231                .with_mutable(false),
232        );
233
234        // Add a move call for each expired proposal
235        for proposal in &expired_proposals {
236            let proposal_id_arg = builder.pure(&proposal.id);
237
238            // Get the type argument for the proposal
239            let type_arg = match &proposal.proposal_type {
240                ProposalType::UpdateConfig => TypeTag::Struct(Box::new(StructTag::new(
241                    hashi_ids.package_id,
242                    Identifier::from_static("update_config"),
243                    Identifier::from_static("UpdateConfig"),
244                    vec![],
245                ))),
246                ProposalType::EnableVersion => TypeTag::Struct(Box::new(StructTag::new(
247                    hashi_ids.package_id,
248                    Identifier::from_static("enable_version"),
249                    Identifier::from_static("EnableVersion"),
250                    vec![],
251                ))),
252                ProposalType::DisableVersion => TypeTag::Struct(Box::new(StructTag::new(
253                    hashi_ids.package_id,
254                    Identifier::from_static("disable_version"),
255                    Identifier::from_static("DisableVersion"),
256                    vec![],
257                ))),
258                ProposalType::Upgrade => TypeTag::Struct(Box::new(StructTag::new(
259                    hashi_ids.package_id,
260                    Identifier::from_static("upgrade"),
261                    Identifier::from_static("Upgrade"),
262                    vec![],
263                ))),
264                ProposalType::EmergencyPause => TypeTag::Struct(Box::new(StructTag::new(
265                    hashi_ids.package_id,
266                    Identifier::from_static("emergency_pause"),
267                    Identifier::from_static("EmergencyPause"),
268                    vec![],
269                ))),
270                ProposalType::AbortReconfig => TypeTag::Struct(Box::new(StructTag::new(
271                    hashi_ids.package_id,
272                    Identifier::from_static("abort_reconfig"),
273                    Identifier::from_static("AbortReconfig"),
274                    vec![],
275                ))),
276                ProposalType::UpdateGuardian => TypeTag::Struct(Box::new(StructTag::new(
277                    hashi_ids.package_id,
278                    Identifier::from_static("update_guardian"),
279                    Identifier::from_static("UpdateGuardian"),
280                    vec![],
281                ))),
282                ProposalType::Unknown(type_name) => {
283                    error!(
284                        "Cannot delete proposal {:?} with unknown type: {}",
285                        proposal.id, type_name
286                    );
287                    continue;
288                }
289            };
290
291            builder.move_call(
292                Function::new(
293                    hashi_ids.package_id,
294                    Identifier::from_static("proposal"),
295                    Identifier::from_static("delete_expired"),
296                )
297                .with_type_args(vec![type_arg]),
298                vec![hashi_arg, proposal_id_arg, clock_arg],
299            );
300        }
301
302        let response = executor.execute(builder).await?;
303        if !response.transaction().effects().status().success() {
304            anyhow::bail!("Transaction failed to delete expired proposals");
305        }
306        info!(
307            "Successfully deleted {} expired proposals",
308            expired_proposals.len()
309        );
310        Ok(())
311    }

314/// Return UTXO IDs whose `spent_epoch` is set — these are spent UTXOs
315/// still present in `utxo_records` that need to be cleaned up on-chain.
316///
317/// This is the pure-data core of [`LeaderService::discover_orphaned_utxo_cleanups`],
318/// extracted so it can be unit-tested without constructing a full `LeaderService`.
319pub(crate) fn find_spent_utxos_pending_cleanup(
320    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
321) -> Vec<UtxoId> {
322    utxo_records
323        .iter()
324        .filter(|(_, record)| record.spent_epoch.is_some())
325        .map(|(id, _)| *id)
326        .collect()
327}

332    use crate::onchain::types::Utxo;

335    /// Helper: build a `UtxoId` from a distinguishing byte and vout.
336    fn utxo_id(byte: u8, vout: u32) -> UtxoId {
337        let mut bytes = [0u8; 32];
338        bytes[0] = byte;
339        UtxoId {
340            txid: BitcoinTxid::new(bytes),
341            vout,
342        }
343    }
344
345    /// Helper: build a `UtxoRecord` with the given `spent_epoch`.
346    fn record(spent_epoch: Option<u64>) -> UtxoRecord {
347        UtxoRecord {
348            utxo: Utxo {
349                id: utxo_id(0, 0),
350                amount: 1_000,
351                derivation_path: None,
352            },
353            produced_by: None,
354            locked_by: None,
355            spent_epoch,
356        }
357    }

359    #[test]
360    fn no_spent_utxos_returns_empty() {
361        let utxo_records: BTreeMap<UtxoId, UtxoRecord> =
362            BTreeMap::from([(utxo_id(1, 0), record(None)), (utxo_id(2, 0), record(None))]);
363
364        let result = find_spent_utxos_pending_cleanup(&utxo_records);
365        assert!(result.is_empty());
366    }
367
368    #[test]
369    fn spent_utxo_found_for_cleanup() {
370        let utxo_records = BTreeMap::from([(utxo_id(1, 0), record(Some(1)))]);
371
372        let result = find_spent_utxos_pending_cleanup(&utxo_records);
373        assert_eq!(result.len(), 1);
374        assert_eq!(result[0], utxo_id(1, 0));
375    }

377    #[test]
378    fn multiple_spent_utxos_found() {
379        let utxo_records = BTreeMap::from([
380            (utxo_id(1, 0), record(Some(1))),
381            (utxo_id(2, 0), record(Some(2))),
382            (utxo_id(3, 0), record(Some(1))),
383        ]);
384
385        let result = find_spent_utxos_pending_cleanup(&utxo_records);
386        assert_eq!(result.len(), 3);
387    }

389    #[test]
390    fn mixed_spent_and_unspent() {
391        let utxo_records = BTreeMap::from([
392            // Unspent
393            (utxo_id(1, 0), record(None)),
394            (utxo_id(2, 0), record(None)),
395            // Spent
396            (utxo_id(3, 0), record(Some(1))),
397            (utxo_id(4, 0), record(Some(2))),
398        ]);
399
400        let result = find_spent_utxos_pending_cleanup(&utxo_records);
401        assert_eq!(result.len(), 2);
402        assert!(result.contains(&utxo_id(3, 0)));
403        assert!(result.contains(&utxo_id(4, 0)));
404    }
