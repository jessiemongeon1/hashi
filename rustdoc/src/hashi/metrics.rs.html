883    pub fn update_onchain_state(&self, state: &crate::onchain::OnchainState) {
884        self.latest_checkpoint_height
885            .set(state.latest_checkpoint_height() as i64);
886        self.latest_checkpoint_timestamp_ms
887            .set(state.latest_checkpoint_timestamp_ms() as i64);
888        self.sui_epoch.set(state.latest_checkpoint_epoch() as i64);
889
890        let guard = state.state();
891        let hashi = guard.hashi();
892
893        self.epoch.set(hashi.committees.epoch() as i64);
894        self.reconfig_in_progress
895            .set(if hashi.committees.pending_epoch_change().is_some() {
896                1
897            } else {
898                0
899            });
900        self.paused.set(if hashi.config.paused() { 1 } else { 0 });
901        self.deposit_queue_size
902            .set(hashi.deposit_queue.requests().len() as i64);
903        let (requested, approved) = hashi
904            .withdrawal_queue
905            .requests()
906            .values()
907            .partition::<Vec<_>, _>(|r| r.status.is_requested());
908        let (signed, pending): (Vec<_>, Vec<_>) = hashi
909            .withdrawal_queue
910            .withdrawal_txns()
911            .values()
912            .partition(|w| w.signatures.is_some());
913        self.withdrawal_queue_size
914            .with_label_values(&["requested"])
915            .set(requested.len() as i64);
916        self.withdrawal_queue_value
917            .with_label_values(&["requested", "BTC"])
918            .set(requested.iter().map(|r| r.btc_amount).sum::<u64>() as i64);
919        self.withdrawal_queue_size
920            .with_label_values(&["approved"])
921            .set(approved.len() as i64);
922        self.withdrawal_queue_value
923            .with_label_values(&["approved", "BTC"])
924            .set(approved.iter().map(|r| r.btc_amount).sum::<u64>() as i64);
925        self.withdrawal_queue_size
926            .with_label_values(&["pending"])
927            .set(pending.len() as i64);
928        self.withdrawal_queue_value
929            .with_label_values(&["pending", "BTC"])
930            .set(
931                pending
932                    .iter()
933                    .flat_map(|w| &w.withdrawal_outputs)
934                    .map(|o| o.amount)
935                    .sum::<u64>() as i64,
936            );
937        self.withdrawal_queue_size
938            .with_label_values(&["signed"])
939            .set(signed.len() as i64);
940        self.withdrawal_queue_value
941            .with_label_values(&["signed", "BTC"])
942            .set(
943                signed
944                    .iter()
945                    .flat_map(|w| &w.withdrawal_outputs)
946                    .map(|o| o.amount)
947                    .sum::<u64>() as i64,
948            );
949        // Track three views of utxo_records:
950        // - available:         all selectable UTXOs (locked_by = None), whether
951        //                      confirmed or not; this is the coin-selection pool
952        // - unconfirmed_change: subset of available whose producing withdrawal
953        //                      has not yet confirmed on Bitcoin (produced_by =
954        //                      Some); useful for gauging mempool chain depth
955        // - locked:            committed to a pending withdrawal, awaiting
956        //                      Bitcoin confirmation (locked_by = Some)
957        let mut available_count = 0i64;
958        let mut unconfirmed_change_count = 0i64;
959        let mut locked_count = 0i64;
960        let mut available_value = 0u64;
961        let mut unconfirmed_change_value = 0u64;
962        let mut locked_value = 0u64;
963        for record in hashi.utxo_pool.utxo_records().values() {
964            if record.locked_by.is_some() {
965                locked_count += 1;
966                locked_value += record.utxo.amount;
967            } else {
968                available_count += 1;
969                available_value += record.utxo.amount;
970                if record.produced_by.is_some() {
971                    unconfirmed_change_count += 1;
972                    unconfirmed_change_value += record.utxo.amount;
973                }
974            }
975        }
976        self.utxo_pool_size
977            .with_label_values(&["available"])
978            .set(available_count);
979        self.utxo_pool_size
980            .with_label_values(&["unconfirmed_change"])
981            .set(unconfirmed_change_count);
982        self.utxo_pool_size
983            .with_label_values(&["locked"])
984            .set(locked_count);
985        self.utxo_pool_value
986            .with_label_values(&["available"])
987            .set(available_value as i64);
988        self.utxo_pool_value
989            .with_label_values(&["unconfirmed_change"])
990            .set(unconfirmed_change_value as i64);
991        self.utxo_pool_value
992            .with_label_values(&["locked"])
993            .set(locked_value as i64);
994        {
995            use crate::onchain::types::ProposalType;
996            let mut counts = std::collections::HashMap::<&str, i64>::new();
997            for proposal in hashi.proposals.active().values() {
998                *counts.entry(proposal.proposal_type.as_str()).or_default() += 1;
999            }
1000            for label in ProposalType::all_labels() {
1001                self.proposals
1002                    .with_label_values(&[label])
1003                    .set(*counts.get(label).unwrap_or(&0));
1004            }
1005        }
1006        self.num_consumed_presigs
1007            .set(hashi.num_consumed_presigs as i64);
1008        for (type_tag, cap) in &hashi.treasury.treasury_caps {
1009            if let sui_sdk_types::TypeTag::Struct(struct_tag) = type_tag {
1010                self.treasury_supply
1011                    .with_label_values(&[struct_tag.name().as_str()])
1012                    .set(cap.supply as i64);
1013            }
1014        }
1015
1016        self.package_version_enabled.reset();
1017        for version in &hashi.config.enabled_versions {
1018            let version_str = version.to_string();
1019            let package_id_str = guard
1020                .package_versions()
1021                .get(version)
1022                .map(|addr| addr.to_string())
1023                .unwrap_or_else(|| "unknown".to_string());
1024            self.package_version_enabled
1025                .with_label_values(&[&version_str, &package_id_str])
1026                .set(1);
1027        }
1028    }

1058fn limiter_outcome_label(
1059    result: &Result<(), crate::guardian_limiter::LocalLimiterError>,
1060) -> &'static str {
1061    use crate::guardian_limiter::LocalLimiterError;
1062    match result {
1063        Ok(()) => GUARDIAN_LIMITER_OUTCOME_SUCCESS,
1064        Err(LocalLimiterError::SeqMismatch { .. }) => GUARDIAN_LIMITER_OUTCOME_SEQ_MISMATCH,
1065        Err(LocalLimiterError::StaleTimestamp { .. }) => GUARDIAN_LIMITER_OUTCOME_STALE_TIMESTAMP,
1066        Err(LocalLimiterError::InsufficientCapacity { .. }) => {
1067            GUARDIAN_LIMITER_OUTCOME_INSUFFICIENT_CAPACITY
1068        }
1069    }
1070}

1072/// Create a metric that measures the uptime from when this metric was constructed.
1073/// The metric is labeled with:
1074/// - 'version': binary version, generally be of the format: 'semver-gitrevision'
1075/// - 'chain_identifier': the identifier of the network which this process is part of
1076pub fn uptime_metric(
1077    version: &'static str,
1078    sui_chain_id: &str,
1079    bitcoin_chain_id: &str,
1080    package_id: &str,
1081    hashi_object_id: &str,
1082) -> Box<dyn prometheus::core::Collector> {
1083    let opts = prometheus::opts!("uptime", "uptime of the node service in seconds")
1084        .variable_label("version")
1085        .variable_label("sui_chain_id")
1086        .variable_label("bitcoin_chain_id")
1087        .variable_label("package_id")
1088        .variable_label("hashi_object_id");
1089
1090    let start_time = std::time::Instant::now();
1091    let uptime = move || start_time.elapsed().as_secs();
1092    let metric = prometheus_closure_metric::ClosureMetric::new(
1093        opts,
1094        prometheus_closure_metric::ValueType::Counter,
1095        uptime,
1096        &[
1097            version,
1098            sui_chain_id,
1099            bitcoin_chain_id,
1100            package_id,
1101            hashi_object_id,
1102        ],
1103    )
1104    .unwrap();
1105
1106    Box::new(metric)
1107}

1111// Creates a new http server that has as a sole purpose to expose
1112// an endpoint that prometheus agent can use to poll for the metrics.
1113pub fn start_prometheus_server(

1115    registry: prometheus::Registry,
1116) -> sui_http::ServerHandle {
1117    let router = axum::Router::new()
1118        .route(METRICS_ROUTE, axum::routing::get(metrics))
1119        .with_state(registry);
1120
1121    sui_http::Builder::new().serve(addr, router).unwrap()
1122}

1124async fn metrics(
1125    axum::extract::State(registry): axum::extract::State<prometheus::Registry>,
1126) -> (http::StatusCode, String) {
1127    let metrics_families = registry.gather();
1128    match prometheus::TextEncoder.encode_to_string(&metrics_families) {
1129        Ok(metrics) => (http::StatusCode::OK, metrics),
1130        Err(error) => (
1131            http::StatusCode::INTERNAL_SERVER_ERROR,
1132            format!("unable to encode metrics: {error}"),
1133        ),
1134    }
1135}
