88/// Client for reading Hashi on-chain state and building/executing transactions.
89///
90/// Uses `OnchainState` for reading on-chain data (committees, proposals, etc.)
91/// and `SuiTxExecutor` for transaction execution when a keypair is configured.
92pub struct HashiClient {
93    /// On-chain state reader from hashi crate
94    onchain_state: OnchainState,
95    /// Hashi package and object IDs
96    hashi_ids: HashiIds,
97    /// Optional executor for signing and submitting transactions
98    executor: Option<SuiTxExecutor>,
99}

101impl HashiClient {
102    /// Create a new client
103    ///
104    /// This scrapes the current on-chain state and optionally sets up
105    /// transaction execution if a keypair is configured.
106    pub async fn new(config: &CliConfig) -> Result<Self> {
107        config.validate()?;
108
109        let hashi_ids = HashiIds {
110            package_id: config.package_id(),
111            hashi_object_id: config.hashi_object_id(),
112        };
113
114        // Create OnchainState which scrapes the current state.
115        // Dropping the service immediately aborts the background watcher task, so the
116        // OnchainState will have the initial scraped state but won't receive live updates.
117        // This is fine for CLI commands for now.
118        let (onchain_state, _service) =
119            OnchainState::new(&config.sui_rpc_url, hashi_ids, None, None, None)
120                .await
121                .context("Failed to initialize on-chain state")?;
122
123        // Try to create executor if keypair is available
124        let executor = match config.load_keypair()? {
125            Some(signer) => {
126                tracing::debug!("Keypair loaded, transaction execution enabled");
127                Some(SuiTxExecutor::new(
128                    onchain_state.client(),
129                    signer,
130                    hashi_ids,
131                ))
132            }
133            None => {
134                tracing::debug!("No keypair configured, transaction execution disabled");
135                None
136            }
137        };
138
139        Ok(Self {
140            onchain_state,
141            hashi_ids,
142            executor,
143        })
144    }

156    /// Execute a transaction built with `TransactionBuilder`.
157    ///
158    /// Returns an error if no keypair is configured.
159    pub async fn execute(
160        &mut self,
161        builder: TransactionBuilder,
162    ) -> Result<ExecuteTransactionResponse> {
163        let executor = self
164            .executor
165            .as_mut()
166            .ok_or_else(|| anyhow::anyhow!("Cannot execute transactions: no keypair configured"))?;
167
168        executor.execute(builder).await
169    }

171    /// Simulate a transaction (dry-run) without executing it.
172    ///
173    /// Returns the estimated gas budget on success.
174    /// This performs the same steps as execute but stops before signing/submitting.
175    pub async fn simulate(&mut self, mut builder: TransactionBuilder) -> Result<SimulationResult> {
176        let executor = self.executor.as_ref().ok_or_else(|| {
177            anyhow::anyhow!("Cannot simulate transactions: no keypair configured")
178        })?;
179
180        let sender = executor.sender();
181        builder.set_sender(sender);
182
183        // Build the transaction - this internally does a dry-run for gas estimation
184        let transaction = builder.build(&mut self.onchain_state.client()).await?;
185
186        Ok(SimulationResult {
187            sender,
188            gas_budget: transaction.gas_payment.budget,
189            gas_price: transaction.gas_payment.price,
190        })
191    }

235    /// Live-fetch the full on-chain `Proposal<T>` (votes + quorum threshold +
236    /// metadata) for a specific proposal via one `list_dynamic_fields` call on
237    /// the proposals bag. Separate from the cached `fetch_proposal` because
238    /// validators don't need these fields in their in-memory state.
239    ///
240    /// The proposal type is derived from the matched child object's
241    /// `object_type` — callers don't pass it, so they can't pass the wrong one.
242    pub async fn fetch_proposal_details(&self, proposal_id: Address) -> Result<ProposalDetails> {
243        use crate::onchain::parse_proposal_type;
244        use crate::onchain::types::ProposalType;
245        use futures::TryStreamExt;
246        use hashi_types::move_types;
247        use sui_rpc::field::FieldMask;
248        use sui_rpc::field::FieldMaskUtil;
249        use sui_rpc::proto::sui::rpc::v2::DynamicField;
250        use sui_rpc::proto::sui::rpc::v2::ListDynamicFieldsRequest;
251
252        // The proposal could live in either the active or executed bag.
253        // Walk active first (the common case), then executed.
254        let (active_id, executed_id) = {
255            let state = self.onchain_state.state();
256            let proposals = &state.hashi().proposals;
257            (proposals.active_id(), proposals.executed_id())
258        };
259        let client = self.onchain_state.client();
260
261        for bag_id in [active_id, executed_id] {
262            let mut stream = Box::pin(
263                client.list_dynamic_fields(
264                    ListDynamicFieldsRequest::default()
265                        .with_parent(bag_id)
266                        .with_page_size(u32::MAX)
267                        .with_read_mask(FieldMask::from_paths([
268                            DynamicField::path_builder().name().finish(),
269                            DynamicField::path_builder().child_object().object_type(),
270                            DynamicField::path_builder()
271                                .child_object()
272                                .contents()
273                                .finish(),
274                        ])),
275                ),
276            );
277
278            while let Some(field) = stream.try_next().await? {
279                // The bag key is BCS-encoded `ID`, which is equivalent to `Address`.
280                let Ok(key) = bcs::from_bytes::<Address>(field.name().value()) else {
281                    continue;
282                };
283                if key != proposal_id {
284                    continue;
285                }
286
287                let object_type_str = field.child_object().object_type();
288                let type_tag: TypeTag = object_type_str
289                    .parse()
290                    .with_context(|| format!("parse object_type {object_type_str:?}"))?;
291                let proposal_type = parse_proposal_type(&type_tag);
292
293                let value_bytes = field.child_object().contents().value();
294                let (creator, votes, quorum_threshold_bps, metadata) = match proposal_type {
295                    ProposalType::UpdateConfig => {
296                        let p: move_types::Proposal<move_types::UpdateConfig> =
297                            bcs::from_bytes(value_bytes).context("deserialize UpdateConfig")?;
298                        (p.creator, p.votes, p.quorum_threshold_bps, p.metadata)
299                    }
300                    ProposalType::EnableVersion => {
301                        let p: move_types::Proposal<move_types::EnableVersion> =
302                            bcs::from_bytes(value_bytes).context("deserialize EnableVersion")?;
303                        (p.creator, p.votes, p.quorum_threshold_bps, p.metadata)
304                    }
305                    ProposalType::DisableVersion => {
306                        let p: move_types::Proposal<move_types::DisableVersion> =
307                            bcs::from_bytes(value_bytes).context("deserialize DisableVersion")?;
308                        (p.creator, p.votes, p.quorum_threshold_bps, p.metadata)
309                    }
310                    ProposalType::Upgrade => {
311                        let p: move_types::Proposal<move_types::Upgrade> =
312                            bcs::from_bytes(value_bytes).context("deserialize Upgrade")?;
313                        (p.creator, p.votes, p.quorum_threshold_bps, p.metadata)
314                    }
315                    ProposalType::EmergencyPause => {
316                        let p: move_types::Proposal<move_types::EmergencyPause> =
317                            bcs::from_bytes(value_bytes).context("deserialize EmergencyPause")?;
318                        (p.creator, p.votes, p.quorum_threshold_bps, p.metadata)
319                    }
320                    ProposalType::AbortReconfig => {
321                        let p: move_types::Proposal<move_types::AbortReconfig> =
322                            bcs::from_bytes(value_bytes).context("deserialize AbortReconfig")?;
323                        (p.creator, p.votes, p.quorum_threshold_bps, p.metadata)
324                    }
325                    ProposalType::UpdateGuardian => {
326                        let p: move_types::Proposal<move_types::UpdateGuardian> =
327                            bcs::from_bytes(value_bytes).context("deserialize UpdateGuardian")?;
328                        (p.creator, p.votes, p.quorum_threshold_bps, p.metadata)
329                    }
330                    ProposalType::Unknown(s) => {
331                        anyhow::bail!("Cannot fetch details for unknown proposal type: {s}")
332                    }
333                };
334                return Ok(ProposalDetails {
335                    creator,
336                    votes,
337                    quorum_threshold_bps,
338                    metadata,
339                });
340            }
341        }
342
343        anyhow::bail!("Proposal {proposal_id} not found in proposals bags")
344    }

385        &self,
386        proposal_id: Address,
387        type_arg: TypeTag,
388    ) -> TransactionBuilder {
389        let mut builder = TransactionBuilder::new();
390
391        let hashi_arg = builder.object(
392            ObjectInput::new(self.hashi_ids.hashi_object_id)
393                .as_shared()
394                .with_mutable(true),
395        );
396        let proposal_id_arg = builder.pure(&proposal_id);

398        builder.move_call(
399            Function::new(
400                self.hashi_ids.package_id,
401                Identifier::from_static("proposal"),
402                Identifier::from_static("remove_vote"),
403            )
404            .with_type_args(vec![type_arg]),
405            vec![hashi_arg, proposal_id_arg],
406        );
407
408        builder
409    }

419    /// Build a transaction to execute a proposal that has reached quorum.
420    ///
421    /// Each proposal type has its own `module::execute(hashi, proposal_id, clock)`
422    /// entry point. This method dispatches to the correct one based on the
423    /// on-chain proposal type.
424    pub fn build_execute_proposal_transaction(
425        &self,
426        proposal_id: Address,
427        proposal_type: &crate::onchain::types::ProposalType,
428    ) -> anyhow::Result<TransactionBuilder> {
429        use crate::onchain::types::ProposalType;
430
431        let module_name = match proposal_type {
432            ProposalType::UpdateConfig => "update_config",
433            ProposalType::EnableVersion => "enable_version",
434            ProposalType::DisableVersion => "disable_version",
435            ProposalType::EmergencyPause => "emergency_pause",
436            ProposalType::AbortReconfig => "abort_reconfig",
437            ProposalType::UpdateGuardian => "update_guardian",
438            ProposalType::Upgrade => {
439                anyhow::bail!(
440                    "Upgrade proposals require the full upgrade flow (execute + publish + finalize)"
441                );
442            }
443            ProposalType::Unknown(s) => {
444                anyhow::bail!("Cannot execute unknown proposal type: {s}");
445            }
446        };
447
448        let mut builder = TransactionBuilder::new();
449        let hashi_arg = builder.object(
450            ObjectInput::new(self.hashi_ids.hashi_object_id)
451                .as_shared()
452                .with_mutable(true),
453        );
454        let proposal_id_arg = builder.pure(&proposal_id);
455        let clock_arg = builder.object(
456            ObjectInput::new(SUI_CLOCK_OBJECT_ID)
457                .as_shared()
458                .with_mutable(false),
459        );
460
461        builder.move_call(
462            Function::new(
463                self.hashi_ids.package_id,
464                Identifier::new(module_name)?,
465                Identifier::from_static("execute"),
466            ),
467            vec![hashi_arg, proposal_id_arg, clock_arg],
468        );
469
470        Ok(builder)
471    }

474/// Build a `TransactionBuilder` for creating a proposal, given `HashiIds` and params.
475///
476/// This is a standalone function so it can be reused outside `HashiClient` (e.g. in tests).
477pub fn build_create_proposal_transaction(
478    hashi_ids: HashiIds,
479    params: CreateProposalParams,
480) -> TransactionBuilder {
481    let mut builder = TransactionBuilder::new();
482
483    let hashi_arg = builder.object(
484        ObjectInput::new(hashi_ids.hashi_object_id)
485            .as_shared()
486            .with_mutable(true),
487    );
488    let clock_arg = builder.object(
489        ObjectInput::new(SUI_CLOCK_OBJECT_ID)
490            .as_shared()
491            .with_mutable(false),
492    );
493
494    match params {
495        CreateProposalParams::Upgrade { digest, metadata } => {
496            let digest_arg = builder.pure(&digest);
497            let metadata_arg = build_metadata(&mut builder, &metadata);
498            builder.move_call(
499                Function::new(
500                    hashi_ids.package_id,
501                    Identifier::from_static("upgrade"),
502                    Identifier::from_static("propose"),
503                ),
504                vec![hashi_arg, digest_arg, metadata_arg, clock_arg],
505            );
506        }
507        CreateProposalParams::UpdateConfig {
508            key,
509            value,
510            metadata,
511        } => {
512            let value_arg = build_config_value(&mut builder, hashi_ids.package_id, &value);
513
514            let key_arg = builder.pure(&key);
515            let metadata_arg = build_metadata(&mut builder, &metadata);
516            builder.move_call(
517                Function::new(
518                    hashi_ids.package_id,
519                    Identifier::from_static("update_config"),
520                    Identifier::from_static("propose"),
521                ),
522                vec![hashi_arg, key_arg, value_arg, metadata_arg, clock_arg],
523            );
524        }
525        CreateProposalParams::UpdateMpcConfig {
526            threshold_bps,
527            max_faulty_bps,
528            weight_reduction_allowed_delta,
529            metadata,
530        } => {
531            let metadata_arg = build_metadata(&mut builder, &metadata);
532            let entries = [
533                ("mpc_threshold_in_basis_points", threshold_bps),
534                ("mpc_max_faulty_in_basis_points", max_faulty_bps),
535                (
536                    "mpc_weight_reduction_allowed_delta",
537                    weight_reduction_allowed_delta,
538                ),
539            ];
540            for (key, value) in entries.into_iter().filter_map(|(k, v)| v.map(|v| (k, v))) {
541                let value_arg = build_config_value(
542                    &mut builder,
543                    hashi_ids.package_id,
544                    &hashi_types::move_types::ConfigValue::U64(value),
545                );
546                let key_arg = builder.pure(&key);
547                builder.move_call(
548                    Function::new(
549                        hashi_ids.package_id,
550                        Identifier::from_static("update_config"),
551                        Identifier::from_static("propose"),
552                    ),
553                    vec![hashi_arg, key_arg, value_arg, metadata_arg, clock_arg],
554                );
555            }
556        }
557        CreateProposalParams::EnableVersion { version, metadata } => {
558            let version_arg = builder.pure(&version);
559            let metadata_arg = build_metadata(&mut builder, &metadata);
560            builder.move_call(
561                Function::new(
562                    hashi_ids.package_id,
563                    Identifier::from_static("enable_version"),
564                    Identifier::from_static("propose"),
565                ),
566                vec![hashi_arg, version_arg, metadata_arg, clock_arg],
567            );
568        }
569        CreateProposalParams::DisableVersion { version, metadata } => {
570            let version_arg = builder.pure(&version);
571            let metadata_arg = build_metadata(&mut builder, &metadata);
572            builder.move_call(
573                Function::new(
574                    hashi_ids.package_id,
575                    Identifier::from_static("disable_version"),
576                    Identifier::from_static("propose"),
577                ),
578                vec![hashi_arg, version_arg, metadata_arg, clock_arg],
579            );
580        }
581        CreateProposalParams::AbortReconfig { epoch, metadata } => {
582            let epoch_arg = builder.pure(&epoch);
583            let metadata_arg = build_metadata(&mut builder, &metadata);
584            builder.move_call(
585                Function::new(
586                    hashi_ids.package_id,
587                    Identifier::from_static("abort_reconfig"),
588                    Identifier::from_static("propose"),
589                ),
590                vec![hashi_arg, epoch_arg, metadata_arg, clock_arg],
591            );
592        }
593        CreateProposalParams::UpdateGuardian {
594            url,
595            public_key,
596            metadata,
597        } => {
598            let url_arg = builder.pure(&url);
599            let public_key_arg = builder.pure(&public_key);
600            let metadata_arg = build_metadata(&mut builder, &metadata);
601            builder.move_call(
602                Function::new(
603                    hashi_ids.package_id,
604                    Identifier::from_static("update_guardian"),
605                    Identifier::from_static("propose"),
606                ),
607                vec![hashi_arg, url_arg, public_key_arg, metadata_arg, clock_arg],
608            );
609        }
610    }
611
612    builder
613}

615/// Build a `config_value::Value` enum via a move call (e.g. `config_value::new_u64(v)`).
616/// Returns the `Argument` holding the constructed `Value`.
617fn build_config_value(
618    builder: &mut TransactionBuilder,
619    package_id: Address,
620    value: &hashi_types::move_types::ConfigValue,
621) -> sui_transaction_builder::Argument {
622    use hashi_types::move_types::ConfigValue;
623
624    let (func_name, arg) = match value {
625        ConfigValue::U64(v) => ("new_u64", builder.pure(v)),
626        ConfigValue::Address(v) => ("new_address", builder.pure(v)),
627        ConfigValue::String(v) => ("new_string", builder.pure(v)),
628        ConfigValue::Bool(v) => ("new_bool", builder.pure(v)),
629        ConfigValue::Bytes(v) => ("new_bytes", builder.pure(v)),
630    };
631
632    builder.move_call(
633        Function::new(
634            package_id,
635            Identifier::from_static("config_value"),
636            Identifier::new(func_name).unwrap(),
637        ),
638        vec![arg],
639    )
640}

642/// Build a `VecMap<String, String>` for proposal metadata via move calls.
643///
644/// Move structs like `VecMap` cannot be passed as pure args in PTBs.
645/// Instead we construct one via `vec_map::empty()` + `vec_map::insert()`.
646fn build_metadata(
647    builder: &mut TransactionBuilder,
648    metadata: &[(String, String)],
649) -> sui_transaction_builder::Argument {
650    let sui_framework = Address::from_static("0x2");
651    let move_stdlib = Address::from_static("0x1");
652
653    let string_type = TypeTag::Struct(Box::new(StructTag::new(
654        move_stdlib,
655        Identifier::from_static("string"),
656        Identifier::from_static("String"),
657        vec![],
658    )));
659
660    // vec_map::empty<String, String>()
661    let map = builder.move_call(
662        Function::new(
663            sui_framework,
664            Identifier::from_static("vec_map"),
665            Identifier::from_static("empty"),
666        )
667        .with_type_args(vec![string_type.clone(), string_type.clone()]),
668        vec![],
669    );
670
671    // vec_map::insert(&mut map, key, value) for each entry
672    for (key, value) in metadata {
673        let key_arg = builder.pure(key);
674        let value_arg = builder.pure(value);
675        builder.move_call(
676            Function::new(
677                sui_framework,
678                Identifier::from_static("vec_map"),
679                Identifier::from_static("insert"),
680            )
681            .with_type_args(vec![string_type.clone(), string_type.clone()]),
682            vec![map, key_arg, value_arg],
683        );
684    }
685
686    map
687}

689/// Build a `proposal::vote<T>` transaction as a standalone. Reusable outside
690/// `HashiClient` — e2e test infra needs to build vote PTBs for every
691/// committee member.
692pub fn build_vote_transaction(
693    hashi_ids: HashiIds,
694    proposal_id: Address,
695    type_arg: TypeTag,
696) -> TransactionBuilder {
697    let mut builder = TransactionBuilder::new();
698    let hashi_arg = builder.object(
699        ObjectInput::new(hashi_ids.hashi_object_id)
700            .as_shared()
701            .with_mutable(true),
702    );
703    let proposal_id_arg = builder.pure(&proposal_id);
704    let clock_arg = builder.object(
705        ObjectInput::new(SUI_CLOCK_OBJECT_ID)
706            .as_shared()
707            .with_mutable(false),
708    );
709
710    builder.move_call(
711        Function::new(
712            hashi_ids.package_id,
713            Identifier::from_static("proposal"),
714            Identifier::from_static("vote"),
715        )
716        .with_type_args(vec![type_arg]),
717        vec![hashi_arg, proposal_id_arg, clock_arg],
718    );
719
720    builder
721}

723/// Get the TypeTag for a proposal type (from on-chain type)
724///
725/// Returns an error if the proposal type is `Unknown`.
726pub fn get_proposal_type_arg(
727    package_id: Address,
728    proposal_type: &crate::onchain::types::ProposalType,
729) -> Result<TypeTag> {
730    use crate::onchain::types::ProposalType;
731
732    let (module, name) = match proposal_type {
733        ProposalType::Upgrade => ("upgrade", "Upgrade"),
734        ProposalType::UpdateConfig => ("update_config", "UpdateConfig"),
735        ProposalType::EnableVersion => ("enable_version", "EnableVersion"),
736        ProposalType::DisableVersion => ("disable_version", "DisableVersion"),
737        ProposalType::EmergencyPause => ("emergency_pause", "EmergencyPause"),
738        ProposalType::AbortReconfig => ("abort_reconfig", "AbortReconfig"),
739        ProposalType::UpdateGuardian => ("update_guardian", "UpdateGuardian"),
740        ProposalType::Unknown(s) => {
741            anyhow::bail!(
742                "Cannot vote on unknown proposal type '{}'. \
743                 This may be a new proposal type not yet supported by this CLI version.",
744                s
745            );
746        }
747    };
748
749    Ok(TypeTag::Struct(Box::new(StructTag::new(
750        package_id,
751        Identifier::new(module).context("Invalid module name")?,
752        Identifier::new(name).context("Invalid type name")?,
753        vec![],
754    ))))
755}
