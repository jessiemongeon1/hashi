58/// BTC that leaves the pool when this txn broadcasts — the amount that
59/// consumes the guardian's limit. Equivalent to
60/// `sum(withdrawal_outputs) + miner_fee`; we use `inputs - change` to
61/// avoid relying on a separate fee field.
62pub(crate) fn withdrawal_limiter_consumption_amount(txn: &WithdrawalTransaction) -> u64 {
63    let inputs: u64 = txn.inputs.iter().map(|u| u.amount).sum();
64    let change: u64 = txn.change_output.as_ref().map_or(0, |c| c.amount);
65    inputs.saturating_sub(change)
66}

68/// Conservative runtime-object budget for withdrawal commit transactions.
69///
70/// Sui's hard object-runtime cache limit is 1000 objects. Empirical
71/// measurements put withdrawal commit transactions at roughly
72/// `selected_utxos + 3 * requests + fixed_overhead` runtime objects. Target
73/// 922 to leave 7.8% headroom below the hard 1000 cap.
74const WITHDRAWAL_COMMIT_RUNTIME_OBJECT_BUDGET: usize = 922;
75const WITHDRAWAL_COMMIT_FIXED_RUNTIME_OBJECTS: usize = 12;
76const WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_REQUEST: usize = 3;
77const WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_INPUT: usize = 1;
78
79/// Full input weight (WU) for a 2-of-2 taproot script-path spend.
80/// TXIN_BASE_WEIGHT (164 WU) + satisfaction (234 WU) = 398 WU (100 vB).
81/// Used in fee validation where we calculate weight directly without
82/// going through Candidate::new().
83const SCRIPT_PATH_2OF2_TXIN_WEIGHT: u64 = 164 + 234;
84
85/// Non-witness fixed overhead for a segwit transaction:
86/// nVersion(4×4) + nLockTime(4×4) = 32 WU, plus the segwit marker/flag (2 WU).
87const TX_FIXED_WEIGHT_WU: u64 = 34;
88
89/// P2TR output weight: TXOUT_BASE(36) + OP_1 OP_PUSHBYTES_32 <32 bytes>(136) = 172 WU.
90const P2TR_OUTPUT_WEIGHT_WU: u64 = 172;
91
92/// P2WPKH output weight: TXOUT_BASE(36) + OP_0 OP_PUSHBYTES_20 <20 bytes>(88) = 124 WU.
93const P2WPKH_OUTPUT_WEIGHT_WU: u64 = 124;
94
95fn safe_withdrawal_commit_max_inputs(request_count: usize, configured_max_inputs: usize) -> usize {
96    let request_objects =
97        request_count.saturating_mul(WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_REQUEST);
98    let runtime_input_budget = WITHDRAWAL_COMMIT_RUNTIME_OBJECT_BUDGET
99        .saturating_sub(WITHDRAWAL_COMMIT_FIXED_RUNTIME_OBJECTS)
100        .saturating_sub(request_objects);
101    let input_budget = runtime_input_budget / WITHDRAWAL_COMMIT_RUNTIME_OBJECTS_PER_INPUT;
102    configured_max_inputs.min(input_budget)
103}
104
105/// Confirm is no longer input-bound: spent UTXOs emit events only (0
106/// objects per input). The confirm transaction's object count is
107/// `requests + ~43` fixed overhead — well within the 1000-object limit
108/// for any practical request count. The commit path is the binding
109/// constraint.
110fn safe_withdrawal_flow_max_inputs(request_count: usize, configured_max_inputs: usize) -> usize {
111    let request_input_budget =
112        request_count.saturating_mul(CoinSelectionParams::DEFAULT_INPUT_BUDGET);
113    configured_max_inputs
114        .min(request_input_budget)
115        .min(safe_withdrawal_commit_max_inputs(
116            request_count,
117            configured_max_inputs,
118        ))
119}

192    #[tracing::instrument(level = "debug", skip_all, fields(bitcoin_txid = %approval.txid))]
193    pub async fn validate_withdrawal_tx_commitment(
194        &self,
195        approval: &WithdrawalTxCommitment,
196    ) -> anyhow::Result<()> {
197        anyhow::ensure!(!approval.request_ids.is_empty(), "No request IDs");
198        anyhow::ensure!(!approval.selected_utxos.is_empty(), "No selected UTXOs");
199        anyhow::ensure!(!approval.outputs.is_empty(), "No outputs");
200
201        // Check for duplicate request IDs
202        let unique_request_ids: std::collections::BTreeSet<_> =
203            approval.request_ids.iter().collect();
204        anyhow::ensure!(
205            unique_request_ids.len() == approval.request_ids.len(),
206            "Duplicate request IDs"
207        );
208
209        // Check for duplicate UTXO IDs
210        let unique_utxo_ids: std::collections::BTreeSet<_> =
211            approval.selected_utxos.iter().collect();
212        anyhow::ensure!(
213            unique_utxo_ids.len() == approval.selected_utxos.len(),
214            "Duplicate UTXO IDs"
215        );
216
217        // 1. Verify each request_id exists and is approved
218        let requests: Vec<WithdrawalRequest> = approval
219            .request_ids
220            .iter()
221            .map(|id| {
222                let request = self
223                    .onchain_state()
224                    .withdrawal_request(id)
225                    .ok_or_else(|| anyhow!("Withdrawal request {id} not found in queue"))?;
226                anyhow::ensure!(
227                    request.status.is_approved(),
228                    "Withdrawal request {id} has not been approved"
229                );
230                Ok(request)
231            })
232            .collect::<anyhow::Result<_>>()?;
233
234        // 2. Verify each selected UTXO exists, is not locked, and collect
235        //    full UTXO data. We look up via utxo_records so we can
236        //    distinguish "missing" from "locked by another withdrawal" and
237        //    also inspect the `produced_by` chain for ancestor depth.
238        let (utxo_records, withdrawal_txns) = {
239            let state = self.onchain_state().state();
240            (
241                state.hashi().utxo_pool.utxo_records().clone(),
242                state.hashi().withdrawal_queue.withdrawal_txns().clone(),
243            )
244        };
245
246        let selected_records: Vec<&UtxoRecord> = approval
247            .selected_utxos
248            .iter()
249            .map(|id| {
250                let record = utxo_records
251                    .get(id)
252                    .ok_or_else(|| anyhow!("UTXO {id:?} not found in the pool"))?;
253                anyhow::ensure!(
254                    record.locked_by.is_none(),
255                    "UTXO {id:?} is locked by pending withdrawal {:?}",
256                    record.locked_by.unwrap()
257                );
258                Ok(record)
259            })
260            .collect::<anyhow::Result<_>>()?;
261
262        // 2b. Verify that no selected UTXO has an unconfirmed ancestor
263        //     chain deeper than Bitcoin Core's relay limit. The limit
264        //     (DEFAULT_ANCESTOR_LIMIT = 25) counts the candidate tx
265        //     itself, so the existing chain must leave room for the
266        //     transaction we are about to construct.
267        for record in &selected_records {
268            let depth = unconfirmed_ancestor_depth(record, &withdrawal_txns, &utxo_records);
269            anyhow::ensure!(
270                depth < MAX_ANCESTOR_DEPTH,
271                "UTXO {:?} has an unconfirmed ancestor chain of depth {} \
272                 which, together with the new transaction, would exceed \
273                 Bitcoin Core's ancestor limit of {}",
274                record.utxo.id,
275                depth,
276                MAX_ANCESTOR_DEPTH,
277            );
278        }
279
280        let selected_utxos: Vec<Utxo> = selected_records.iter().map(|r| r.utxo.clone()).collect();
281
282        // 3. Verify output count: one per request, plus at most one change output
283        let request_count = requests.len();
284        let output_count = approval.outputs.len();
285        anyhow::ensure!(
286            output_count == request_count || output_count == request_count + 1,
287            "Expected {} or {} outputs, got {}",
288            request_count,
289            request_count + 1,
290            output_count
291        );
292
293        // 4. Compute miner fee and verify the per-user fee split
294        let input_total: u64 = selected_utxos.iter().map(|u| u.amount).sum();
295        let output_total: u64 = approval.outputs.iter().map(|o| o.amount).sum();
296        anyhow::ensure!(
297            input_total >= output_total,
298            "Inputs ({input_total}) < outputs ({output_total})"
299        );
300        let fee = input_total - output_total;
301
302        let per_user_miner_fee = fee / request_count as u64;
303
304        // Verify per-user miner fee does not exceed worst-case budget
305        let max_network_fee = self.onchain_state().worst_case_network_fee();
306        anyhow::ensure!(
307            per_user_miner_fee <= max_network_fee,
308            "Per-user miner fee {} sats exceeds worst-case budget {} sats",
309            per_user_miner_fee,
310            max_network_fee
311        );
312
313        // Verify each positional withdrawal output matches the expected amount and address.
314        // request.btc_amount is the full withdrawal amount.
315        for (i, request) in requests.iter().enumerate() {
316            let output = &approval.outputs[i];
317            let expected_amount = request.btc_amount - per_user_miner_fee;
318            anyhow::ensure!(
319                expected_amount >= utxo_pool::TR_DUST_RELAY_MIN_VALUE,
320                "Withdrawal output {} sats is below dust threshold {} sats",
321                expected_amount,
322                utxo_pool::TR_DUST_RELAY_MIN_VALUE
323            );
324            anyhow::ensure!(
325                output.amount == expected_amount,
326                "Output {i} amount {} does not match expected {} for request {:?}",
327                output.amount,
328                expected_amount,
329                request.id
330            );
331            anyhow::ensure!(
332                output.bitcoin_address == request.bitcoin_address,
333                "Output {i} address does not match request {:?}",
334                request.id
335            );
336        }
337
338        // 5. Verify change output (if present) goes to hashi root pubkey
339        if output_count == request_count + 1 {
340            let change_output = &approval.outputs[request_count];
341            let hashi_pubkey = self.get_hashi_pubkey()?;
342            let expected_address =
343                witness_program_from_address(&self.get_deposit_address(&hashi_pubkey, None)?)?;
344            anyhow::ensure!(
345                change_output.bitcoin_address == expected_address,
346                "Change output does not go to hashi root pubkey"
347            );
348            anyhow::ensure!(
349                change_output.amount >= utxo_pool::TR_DUST_RELAY_MIN_VALUE,
350                "Change output {} sats is below dust threshold {} sats",
351                change_output.amount,
352                utxo_pool::TR_DUST_RELAY_MIN_VALUE
353            );
354        }
355
356        // 6. Validate fee is reasonable.
357        //
358        // TODO: When spending unconfirmed change UTXOs the effective fee
359        // rate that matters for mining is the *package* fee rate across
360        // the entire ancestor chain (CPFP). This check currently
361        // evaluates the transaction in isolation, which may over- or
362        // under-estimate the true cost to get the package mined. A
363        // future revision should compute the aggregate ancestor weight
364        // and fees and validate the package fee rate instead.
365        {
366            // Estimate transaction weight.
367            let num_inputs = selected_utxos.len() as u64;
368            let input_weight = SCRIPT_PATH_2OF2_TXIN_WEIGHT * num_inputs;
369            let output_weight: u64 = approval
370                .outputs
371                .iter()
372                .map(|o| output_weight_for_address(&o.bitcoin_address))
373                .collect::<anyhow::Result<Vec<_>>>()?
374                .iter()
375                .sum();
376            let tx_weight = Weight::from_wu(TX_FIXED_WEIGHT_WU + input_weight + output_weight);
377
378            let min_fee_rate = FeeRate::from_sat_per_vb_unchecked(1);
379
380            // Fee must be at least the minimum relay fee (1 sat/vB).
381            let min_fee = min_fee_rate
382                .fee_wu(tx_weight)
383                .map(|a| a.to_sat())
384                .unwrap_or(0);
385            anyhow::ensure!(
386                fee >= min_fee,
387                "Fee {fee} sats is below minimum relay fee {min_fee} sats"
388            );
389
390            // Fee ceiling: clamp the fee rate from our Bitcoin node to
391            // a floor of 1 sat/vB, then cap at the tolerance multiplier.
392            let kyoto_fee_rate = self
393                .btc_monitor()
394                .get_recent_fee_rate(self.config.withdrawal_fee_conf_target())
395                .await?;
396            let clamped_fee_rate = std::cmp::max(kyoto_fee_rate, min_fee_rate);
397            let estimated_fee = clamped_fee_rate
398                .fee_wu(tx_weight)
399                .map(|a| a.to_sat())
400                .unwrap_or(0);
401            let max_fee = estimated_fee.saturating_mul(FEE_RATE_TOLERANCE_MULTIPLIER);
402            anyhow::ensure!(
403                fee <= max_fee,
404                "Fee {fee} sats exceeds maximum allowed {max_fee} sats \
405                 ({FEE_RATE_TOLERANCE_MULTIPLIER}x the clamped estimate of \
406                 {estimated_fee} sats at {clamped_fee_rate})"
407            );
408        }
409
410        // 7. Rebuild unsigned tx and verify txid matches.
411        let tx = self.build_unsigned_withdrawal_tx(&selected_utxos, &approval.outputs)?;
412        let expected_txid = BitcoinTxid::from(tx.compute_txid());
413        anyhow::ensure!(
414            approval.txid == expected_txid,
415            "Txid mismatch: approval has {:?}, rebuilt tx has {:?}",
416            approval.txid,
417            expected_txid
418        );
419
420        Ok(())
421    }

452        withdrawal_txn_id: &Address,

461                "Guardian withdrawal timestamp {timestamp_secs} is {drift}s away from \
462                 local checkpoint {latest_checkpoint_secs} (tolerance: {GUARDIAN_TIMESTAMP_TOLERANCE_SECS}s)"
463            );
464        }
465
466        let txn = self
467            .onchain_state()
468            .withdrawal_txn(withdrawal_txn_id)
469            .ok_or_else(|| {
470                anyhow!("WithdrawalTransaction {withdrawal_txn_id} not found on-chain")
471            })?;
472
473        let guardian_request = build_guardian_withdrawal_request(self, &txn, timestamp_secs, seq)?;
474
475        self.sign_message_proto(&guardian_request)
476    }

480    #[tracing::instrument(level = "info", skip_all, fields(withdrawal_id = %message.withdrawal_id))]
481    pub fn validate_and_sign_withdrawal_tx_signing(
482        &self,
483        message: &WithdrawalTxSigning,
484    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
485        let txn = self
486            .onchain_state()
487            .withdrawal_txn(&message.withdrawal_id)
488            .ok_or_else(|| {
489                anyhow!(
490                    "WithdrawalTransaction {} not found on-chain",
491                    message.withdrawal_id
492                )
493            })?;
494
495        anyhow::ensure!(
496            txn.signatures.is_none(),
497            "WithdrawalTransaction {} is already signed",
498            message.withdrawal_id
499        );
500
501        anyhow::ensure!(
502            message.request_ids == txn.request_ids,
503            "Request IDs mismatch for WithdrawalTransaction {}",
504            message.withdrawal_id
505        );
506
507        anyhow::ensure!(
508            message.signatures.len() == txn.inputs.len(),
509            "Signature count ({}) does not match input count ({}) for WithdrawalTransaction {}",
510            message.signatures.len(),
511            txn.inputs.len(),
512            message.withdrawal_id
513        );
514
515        let tx = self.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs())?;
516        let signing_messages = self.withdrawal_signing_messages(&tx, &txn.inputs)?;
517        let hashi_pubkey = self.get_hashi_pubkey()?;
518
519        for (i, (sig_bytes, sighash)) in message
520            .signatures
521            .iter()
522            .zip(signing_messages.iter())
523            .enumerate()
524        {
525            let arr: &[u8; 64] = sig_bytes.as_slice().try_into().map_err(|_| {
526                anyhow!(
527                    "Signature {i} is not 64 bytes for WithdrawalTransaction {}",
528                    message.withdrawal_id
529                )
530            })?;
531            let sig = SchnorrSignature::from_byte_array(arr)
532                .map_err(|e| anyhow!("Invalid Schnorr signature at input {i}: {e}"))?;
533
534            let input_pubkey =
535                self.deposit_pubkey(&hashi_pubkey, txn.inputs[i].derivation_path.as_ref())?;
536            let schnorr_pk = SchnorrPublicKey::from_byte_array(&input_pubkey.serialize())
537                .map_err(|e| anyhow!("Failed to convert pubkey for input {i}: {e}"))?;
538
539            schnorr_pk
540                .verify(sighash, &sig)
541                .map_err(|e| anyhow!("Signature verification failed for input {i}: {e}"))?;
542        }
543
544        self.sign_message_proto(message)
545    }

549    /// Proto-format BLS signing helper for gRPC responses.
550    fn sign_message_proto<T: serde::Serialize>(
551        &self,
552        message: &T,
553    ) -> anyhow::Result<hashi_types::proto::MemberSignature> {
554        let epoch = self.onchain_state().epoch();
555        let validator_address = self
556            .config
557            .validator_address()
558            .map_err(|e| anyhow!("No validator address configured: {e}"))?;
559        let committee = self
560            .onchain_state()
561            .state()
562            .hashi()
563            .committees
564            .committees()
565            .get(&epoch)
566            .cloned()
567            .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?;
568        let private_key =
569            self.find_signing_key_for_committee(&committee, validator_address, epoch)?;
570        let public_key_bytes = private_key.public_key().as_bytes().to_vec().into();
571        let signature_bytes = private_key
572            .sign(epoch, validator_address, message)
573            .signature()
574            .as_bytes()
575            .to_vec()
576            .into();
577
578        Ok(hashi_types::proto::MemberSignature {

600        &self,
601        withdrawal_txn_id: &Address,

603        crate::onchain::types::WithdrawalTransaction,

626    /// Produce MPC Schnorr signatures for an unsigned withdrawal transaction.
627    #[tracing::instrument(
628        level = "debug",
629        skip_all,
630        fields(withdrawal_txn_id = %txn.id, input_count = txn.inputs.len()),
631    )]
632    async fn mpc_sign_withdrawal_tx(
633        &self,
634        txn: &crate::onchain::types::WithdrawalTransaction,
635        unsigned_tx: &bitcoin::Transaction,
636        expected_limiter_seq: Option<u64>,
637    ) -> anyhow::Result<Vec<SchnorrSignature>> {
638        let onchain_state = self.onchain_state().clone();
639        let epoch = onchain_state.epoch();
640        if txn.epoch != epoch {
641            anyhow::bail!(
642                "Stale presig assignment: pending withdrawal {} has epoch {}, current is {}. \
643                 Either the leader hasn't called allocate_presigs_for_withdrawal_txn yet, \
644                 or this node's on-chain state is behind.",
645                txn.id,
646                txn.epoch,
647                epoch,
648            );
649        }
650        // Validate-only: the watcher advances the limiter on
651        // `WithdrawalSignedEvent`, never from this path.
652        match (self.local_limiter(), expected_limiter_seq) {
653            (Some(limiter), Some(expected_seq)) => {
654                let amount_sats = withdrawal_limiter_consumption_amount(txn);
655                let timestamp_secs = txn.timestamp_ms / 1000;
656                let result = limiter.validate_consume(expected_seq, timestamp_secs, amount_sats);
657                self.metrics.record_limiter_validate(
658                    &result,
659                    crate::metrics::GUARDIAN_LIMITER_CALLSITE_MPC_SIGNING,
660                );
661                result.map_err(|e| anyhow!("Limiter rejected withdrawal {}: {e}", txn.id))?;
662            }
663            (None, None) => {}
664            (Some(_), None) => anyhow::bail!(
665                "Local limiter is configured but request for withdrawal {} lacks expected_limiter_seq",
666                txn.id
667            ),
668            (None, Some(_)) => anyhow::bail!(
669                "Request for withdrawal {} carries expected_limiter_seq but local limiter is not configured",
670                txn.id
671            ),
672        }
673        let p2p_channel =
674            RpcP2PChannel::new(onchain_state, epoch, crate::metrics::MPC_LABEL_SIGNING);
675        let signing_manager = self.signing_manager_for(epoch).ok_or_else(|| {
676            anyhow::anyhow!(
677                "SigningManager not available for epoch {epoch}; \
678                 reconciliation may be catching up"
679            )
680        })?;
681        let beacon = S::from_bytes_mod_order(&txn.randomness);
682        let signing_messages = self.withdrawal_signing_messages(unsigned_tx, &txn.inputs)?;
683        let mut signatures_by_input = Vec::with_capacity(signing_messages.len());
684        for (input_index, message) in signing_messages.iter().enumerate() {
685            let request_id = withdrawal_input_signing_request_id(&txn.id, input_index as u32);
686            let derivation_address = txn
687                .inputs
688                .get(input_index)
689                .and_then(|input| input.derivation_path.as_ref().map(|path| path.into_inner()));
690            let sign_start = std::time::Instant::now();
691            let global_presig_index = txn.presig_start_index + input_index as u64;
692            let sign_result = signing_manager
693                .sign(
694                    &p2p_channel,
695                    request_id,
696                    message,
697                    global_presig_index,
698                    &beacon,
699                    derivation_address.as_ref(),
700                    WITHDRAWAL_SIGNING_TIMEOUT,
701                    &self.metrics,
702                )
703                .await;
704            let sign_duration = sign_start.elapsed().as_secs_f64();
705
706            match &sign_result {
707                Ok(_) => {
708                    self.metrics
709                        .mpc_sign_duration_seconds
710                        .with_label_values(&["success"])
711                        .observe(sign_duration);
712                    self.metrics
713                        .presig_pool_remaining
714                        .set(signing_manager.presignatures_remaining() as i64);
715                }
716                Err(e) => {
717                    self.metrics
718                        .mpc_sign_duration_seconds
719                        .with_label_values(&["failure"])
720                        .observe(sign_duration);
721                    let reason = match e {
722                        crate::mpc::types::SigningError::Timeout { .. } => "timeout",
723                        crate::mpc::types::SigningError::PoolExhausted => "pool_exhausted",
724                        crate::mpc::types::SigningError::TooManyInvalidSignatures { .. } => {
725                            "too_many_invalid"
726                        }
727                        crate::mpc::types::SigningError::CryptoError(_) => "crypto_error",
728                        _ => "other",
729                    };
730                    self.metrics
731                        .mpc_sign_failures_total
732                        .with_label_values(&[reason])
733                        .inc();
734                }
735            }
736
737            let signature = sign_result.map_err(|e| {
738                anyhow!("Failed to sign withdrawal transaction input {input_index}: {e}")
739            })?;
740
741            signatures_by_input.push(signature);
742        }
743        Ok(signatures_by_input)
744    }

746    pub(crate) fn withdrawal_signing_messages(
747        &self,
748        unsigned_tx: &bitcoin::Transaction,
749        inputs: &[Utxo],
750    ) -> anyhow::Result<Vec<[u8; 32]>> {
751        let hashi_pubkey = self.get_hashi_pubkey()?;
752        let spend_inputs = inputs
753            .iter()
754            .map(|input| {
755                let pubkey = self.deposit_pubkey(&hashi_pubkey, input.derivation_path.as_ref())?;
756                let address = self.bitcoin_address_from_pubkey(&pubkey);
757                let (_, _, leaf_hash) =
758                    bitcoin_utils::single_key_taproot_script_path_spend_artifacts(&pubkey);
759                Ok((
760                    TxOut {
761                        value: Amount::from_sat(input.amount),
762                        script_pubkey: address.script_pubkey(),
763                    },
764                    leaf_hash,
765                ))
766            })
767            .collect::<anyhow::Result<Vec<_>>>()?;
768        let prevouts = spend_inputs
769            .iter()
770            .map(|(txout, _)| txout.clone())
771            .collect::<Vec<_>>();
772        let leaf_hashes = spend_inputs
773            .iter()
774            .map(|(_, leaf_hash)| *leaf_hash)
775            .collect::<Vec<TapLeafHash>>();
776
777        (0..inputs.len())
778            .map(|input_index| {
779                let mut sighasher = SighashCache::new(unsigned_tx);
780                let sighash = sighasher
781                    .taproot_script_spend_signature_hash(
782                        input_index,
783                        &Prevouts::All(&prevouts),
784                        leaf_hashes[input_index],
785                        TapSighashType::Default,
786                    )
787                    .map_err(|e| {
788                        anyhow!("Failed to construct taproot script spend sighash: {e}")
789                    })?;
790                Ok(*sighash.as_byte_array())
791            })
792            .collect()
793    }

797    /// Build an unsigned Bitcoin transaction for a withdrawal. This is used both
798    /// by the leader when initially crafting the tx, and by validators when
799    /// verifying that a proposed `WithdrawalTxCommitment` produces the expected txid.
800    pub fn build_unsigned_withdrawal_tx(
801        &self,
802        selected_utxos: &[Utxo],
803        outputs: &[OutputUtxo],
804    ) -> anyhow::Result<bitcoin::Transaction> {
805        let inputs: Vec<bitcoin::TxIn> = selected_utxos
806            .iter()
807            .map(|utxo| bitcoin::TxIn {
808                previous_output: bitcoin::OutPoint {
809                    txid: utxo.id.txid.into(),
810                    vout: utxo.id.vout,
811                },
812                script_sig: bitcoin::ScriptBuf::default(),
813                sequence: bitcoin::Sequence::ENABLE_RBF_NO_LOCKTIME,
814                witness: bitcoin::Witness::default(),
815            })
816            .collect();
817
818        let tx_outputs: Vec<bitcoin::TxOut> = outputs
819            .iter()
820            .map(|output| {
821                let script_pubkey = script_pubkey_from_raw_address(&output.bitcoin_address)
822                    .expect("invalid bitcoin address in output");
823                bitcoin::TxOut {
824                    value: bitcoin::Amount::from_sat(output.amount),
825                    script_pubkey,
826                }
827            })
828            .collect();
829
830        Ok(bitcoin_utils::construct_tx(inputs, tx_outputs))
831    }

833    /// Build a withdrawal commitment for a batch of approved requests: select
834    /// UTXOs using the batching-aware coin selection algorithm, build the
835    /// unsigned BTC tx, and return a `WithdrawalTxCommitment` covering the
836    /// selected requests.
837    #[tracing::instrument(level = "debug", skip_all, fields(request_count = requests.len()))]
838    pub async fn build_withdrawal_tx_commitment(
839        &self,
840        requests: &[WithdrawalRequest],
841    ) -> Result<WithdrawalTxCommitment, WithdrawalCommitmentError> {
842        // Fetch current fee rate from the Bitcoin node, clamped to a high
843        // fee rate threshold to avoid overpaying during fee spikes.
844        let kyoto_fee_rate = self
845            .btc_monitor()
846            .get_recent_fee_rate(self.config.withdrawal_fee_conf_target())
847            .await
848            .map_err(|e| WithdrawalCommitmentError::FeeEstimateFailed(anyhow!(e)))?;
849        let min_fee_rate = CoinSelectionParams::DEFAULT_MIN_FEE_RATE;
850        let max_fee_rate = CoinSelectionParams::DEFAULT_HIGH_FEE_RATE_THRESHOLD;
851        let fee_rate = kyoto_fee_rate.clamp(min_fee_rate, max_fee_rate);
852
853        let hashi_pubkey = self
854            .get_hashi_pubkey()
855            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
856        let change_address = self
857            .get_deposit_address(&hashi_pubkey, None)
858            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
859
860        let configured_max_inputs = CoinSelectionParams::DEFAULT_MAX_INPUTS;
861        let configured_long_term_fee_rate = CoinSelectionParams::DEFAULT_LONG_TERM_FEE_RATE;
862        let configured_max_requests = self.config.withdrawal_max_batch_size().min(requests.len());
863
864        // Snapshot both maps under a single read-lock so they are always
865        // mutually consistent (e.g., a WithdrawalConfirmedEvent cannot update
866        // one map but not the other between the two reads).
867        let (withdrawal_txns, utxo_records) = {
868            let state = self.onchain_state().state();
869            (
870                state.hashi().withdrawal_queue.withdrawal_txns().clone(),
871                state.hashi().utxo_pool.utxo_records().clone(),
872            )
873        };
874
875        // Query Bitcoin in parallel for the confirmation count of every
876        // pending withdrawal so we can accurately fill AncestorTx::confirmations
877        // instead of always hardcoding 0.
878        let tx_confirmations = fetch_withdrawal_tx_confirmations(self, &withdrawal_txns).await;
879
880        // Map available (unlocked) UTXOs to UtxoCandidates.
881        let candidates: Vec<UtxoCandidate> = utxo_records
882            .values()
883            .filter(|r| r.locked_by.is_none())
884            .map(|r| {
885                let status =
886                    build_utxo_status(self, r, &withdrawal_txns, &tx_confirmations, &utxo_records);
887                UtxoCandidate {
888                    id: r.utxo.id,
889                    amount: r.utxo.amount,
890                    spend_path: SpendPath::TaprootScriptPath2of2,
891                    status,
892                }
893            })
894            .collect();
895
896        // Map on-chain WithdrawalRequests to the coin-selector view.
897        // btc_amount is the full withdrawal amount.
898        let mapped_requests: Vec<utxo_pool::WithdrawalRequest> = requests
899            .iter()
900            .map(|r| utxo_pool::WithdrawalRequest {
901                id: r.id,
902                recipient: r.bitcoin_address.clone(),
903                amount: r.btc_amount,
904                timestamp_ms: r.timestamp_ms,
905            })
906            .collect();
907
908        let mut last_selection_error = None;
909        let mut result = None;
910        for request_count in (1..=configured_max_requests).rev() {
911            let max_inputs = safe_withdrawal_flow_max_inputs(request_count, configured_max_inputs);
912            if max_inputs == 0 {
913                continue;
914            }
915
916            let params = CoinSelectionParams {
917                max_inputs,
918                long_term_fee_rate: configured_long_term_fee_rate,
919                max_fee_per_request: self.onchain_state().worst_case_network_fee(),
920                max_withdrawal_requests: request_count,
921                max_mempool_chain_depth: self.config.max_mempool_chain_depth(),
922                ..CoinSelectionParams::new(change_address.clone())
923            };
924
925            match utxo_pool::select_coins(&candidates, &mapped_requests, &params, fee_rate) {
926                Ok(selection) => {
927                    if request_count < configured_max_requests {
928                        tracing::info!(
929                            selected_requests = selection.selected_requests.len(),
930                            selected_inputs = selection.inputs.len(),
931                            configured_max_requests,
932                            configured_max_inputs,
933                            max_inputs,
934                            "Reduced withdrawal batch to stay within Sui commit limits",
935                        );
936                    }
937                    result = Some(selection);
938                    break;
939                }
940                Err(e) => last_selection_error = Some(e),
941            }
942        }
943
944        let result = result.ok_or_else(|| {
945            WithdrawalCommitmentError::UtxoSelectionFailed(anyhow!(
946                last_selection_error
947                    .map(|e| e.to_string())
948                    .unwrap_or_else(|| "no withdrawal request count fits Sui commit limits".into())
949            ))
950        })?;
951
952        // Build outputs: one per selected request (net amount already deducted),
953        // plus an optional change output.
954        let mut outputs: Vec<OutputUtxo> = result
955            .withdrawal_outputs
956            .iter()
957            .map(|o| OutputUtxo {
958                amount: o.amount,
959                bitcoin_address: o.recipient.clone(),
960            })
961            .collect();
962
963        if let Some(change_amount) = result.change {
964            outputs.push(OutputUtxo {
965                amount: change_amount,
966                bitcoin_address: witness_program_from_address(&change_address)
967                    .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?,
968            });
969        }
970
971        let selected_utxos: Vec<UtxoId> = result.inputs.iter().map(|u| u.id).collect();
972        let request_ids: Vec<Address> = result.selected_requests.iter().map(|r| r.id).collect();
973
974        // Resolve UtxoCandidates back to full Utxo objects for tx building.
975        let selected_input_utxos: Vec<Utxo> = result
976            .inputs
977            .iter()
978            .map(|c| self.onchain_state().active_utxo(&c.id))
979            .collect::<Option<Vec<_>>>()
980            .ok_or_else(|| {
981                WithdrawalCommitmentError::BtcTxBuildFailed(anyhow!(
982                    "a selected UTXO disappeared from the pool between selection and tx build"
983                ))
984            })?;
985
986        let tx = self
987            .build_unsigned_withdrawal_tx(&selected_input_utxos, &outputs)
988            .map_err(WithdrawalCommitmentError::BtcTxBuildFailed)?;
989        let txid = BitcoinTxid::from(tx.compute_txid());
990
991        Ok(WithdrawalTxCommitment {
992            request_ids,
993            selected_utxos,
994            outputs,
995            txid,
996        })
997    }

999    /// Run AML/Sanctions checks for a withdrawal request.
1000    /// If no screener client is configured, checks are skipped.
1001    #[tracing::instrument(level = "debug", skip_all, fields(request_id = %request.id))]
1002    pub(crate) async fn screen_withdrawal(
1003        &self,
1004        request: &WithdrawalRequest,
1005    ) -> Result<(), WithdrawalApprovalError> {
1006        let Some(screener) = self.screener_client() else {
1007            tracing::debug!("AML checks skipped: no screener configured");
1008            return Ok(());
1009        };
1010
1011        // Source: Sui tx digest (base58 string)
1012        let source_tx_hash = request.sui_tx_digest.to_string();
1013
1014        // Destination: Bitcoin address (raw witness bytes -> bech32 string)
1015        let destination_address = self
1016            .bitcoin_address_string_from_raw(&request.bitcoin_address)
1017            .map_err(WithdrawalApprovalError::NeverRetry)?;
1018
1019        let approved = screener
1020            .approve_withdrawal(
1021                &source_tx_hash,
1022                &destination_address,
1023                self.config.sui_chain_id(),
1024                self.config.bitcoin_chain_id(),
1025            )
1026            .await
1027            .map_err(|e| WithdrawalApprovalError::AmlServiceError(anyhow!(e)))?;
1028
1029        if !approved {
1030            return Err(WithdrawalApprovalError::NeverRetry(anyhow!(
1031                "AML checks failed for withdrawal request {:?} to {}",
1032                request.id,
1033                destination_address,
1034            )));
1035        }
1036
1037        Ok(())
1038    }

1040    /// Convert raw witness program bytes to a human-readable Bitcoin address string.
1041    fn bitcoin_address_string_from_raw(&self, address_bytes: &[u8]) -> anyhow::Result<String> {
1042        let version = match address_bytes.len() {
1043            32 => WitnessVersion::V1,
1044            20 => WitnessVersion::V0,
1045            len => anyhow::bail!("Unsupported bitcoin address length: {len}"),
1046        };
1047        let program = WitnessProgram::new(version, address_bytes)
1048            .map_err(|e| anyhow!("Invalid witness program: {e}"))?;
1049        let script = bitcoin::ScriptBuf::new_witness_program(&program);
1050        let address = bitcoin::Address::from_script(&script, self.config.bitcoin_network())
1051            .map_err(|e| anyhow!("Failed to convert script to address: {e}"))?;
1052        Ok(address.to_string())
1053    }

1177/// Query Bitcoin in parallel for the confirmation count of every pending
1178/// withdrawal transaction. Returns a map from withdrawal ID to confirmation
1179/// count. Withdrawals that are in the mempool, not found, or whose RPC call
1180/// fails are mapped to 0 (treated as unconfirmed).
1181async fn fetch_withdrawal_tx_confirmations(
1182    hashi: &Hashi,
1183    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1184) -> HashMap<Address, u32> {
1185    let futures: Vec<_> = withdrawal_txns
1186        .iter()
1187        .map(|(id, txn)| async {
1188            let btc_txid = txn.txid.into();
1189            let confs = match hashi.btc_monitor().get_transaction_status(btc_txid).await {
1190                Ok(TxStatus::Confirmed { confirmations }) => confirmations,
1191                // Mempool, not found, or RPC error — treat as unconfirmed.
1192                _ => 0,
1193            };
1194            (*id, confs)
1195        })
1196        .collect();
1197    futures::future::join_all(futures)
1198        .await
1199        .into_iter()
1200        .collect()
1201}

1203/// Build the [`UtxoStatus`] for a UTXO record using a pre-fetched snapshot.
1204///
1205/// For confirmed UTXOs (`produced_by = None`) this is simply
1206/// [`UtxoStatus::Confirmed`]. For unconfirmed change outputs
1207/// (`produced_by = Some(withdrawal_id)`) we walk the full ancestor chain
1208/// so that CPFP weight and mempool depth are accurately computed even for
1209/// multi-level chains. If the producing withdrawal has already been removed
1210/// from `withdrawal_txns` (confirmed and cleared), we promote the UTXO
1211/// to `Confirmed` — it is safe to spend immediately.
1212fn build_utxo_status(
1213    hashi: &Hashi,
1214    record: &UtxoRecord,
1215    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1216    tx_confirmations: &HashMap<Address, u32>,
1217    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1218) -> UtxoStatus {
1219    let Some(producing_id) = record.produced_by else {
1220        return UtxoStatus::Confirmed;
1221    };
1222
1223    let chain = build_ancestor_chain(
1224        hashi,
1225        producing_id,
1226        withdrawal_txns,
1227        tx_confirmations,
1228        utxo_records,
1229    );
1230
1231    if chain.is_empty() {
1232        // The producing withdrawal was confirmed and removed from
1233        // withdrawal_txns. The UTXO is safe to spend.
1234        UtxoStatus::Confirmed
1235    } else {
1236        UtxoStatus::Pending { chain }
1237    }
1238}

1240/// Maximum ancestor depth permitted by Bitcoin Core's relay policy
1241/// (`DEFAULT_ANCESTOR_LIMIT = 25`). Bitcoin Core counts the candidate
1242/// transaction itself in the ancestor set, so a UTXO whose existing
1243/// unconfirmed ancestor depth is already `MAX_ANCESTOR_DEPTH - 1` is
1244/// the deepest we can safely spend.
1245pub const MAX_ANCESTOR_DEPTH: usize = 25;
1246
1247/// Count the number of unconfirmed ancestors for a UTXO record by walking
1248/// the `produced_by` chain. Every ancestor that still appears in
1249/// `withdrawal_txns` is conservatively treated as unconfirmed (we skip
1250/// querying Bitcoin for actual confirmation counts). This is used during
1251/// commitment validation to reject UTXOs whose ancestor chain would exceed
1252/// Bitcoin Core's relay limit.
1253///
1254/// The walk is a BFS over the ancestor DAG. Each item on the queue is a
1255/// `(producing_withdrawal_id, depth)` pair. We track the maximum depth
1256/// seen across all branches.
1257fn unconfirmed_ancestor_depth(
1258    record: &UtxoRecord,
1259    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1260    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1261) -> usize {
1262    let Some(producing_id) = record.produced_by else {
1263        return 0;
1264    };
1265
1266    let mut max_depth: usize = 0;
1267    let mut queue = std::collections::VecDeque::new();
1268    queue.push_back((producing_id, 1usize));
1269
1270    while let Some((wid, depth)) = queue.pop_front() {
1271        if depth > MAX_ANCESTOR_DEPTH {
1272            return depth;
1273        }
1274
1275        let Some(txn) = withdrawal_txns.get(&wid) else {
1276            // The producing withdrawal has been confirmed and removed;
1277            // it does not contribute to the unconfirmed chain.
1278            continue;
1279        };
1280
1281        max_depth = std::cmp::max(max_depth, depth);
1282
1283        // Enqueue any inputs that are themselves unconfirmed change
1284        // outputs of an earlier withdrawal.
1285        for input_utxo in &txn.inputs {
1286            if let Some(input_record) = utxo_records.get(&input_utxo.id)
1287                && let Some(parent_id) = input_record.produced_by
1288            {
1289                queue.push_back((parent_id, depth + 1));
1290            }
1291        }
1292    }
1293
1294    max_depth
1295}

1297/// Build the ancestor chain for a UTXO produced by `producing_id`. Each
1298/// unconfirmed ancestor that still appears in `withdrawal_txns` gets
1299/// one [`AncestorTx`] entry with its confirmation count, weight, and fee.
1300/// The walk is a BFS over the ancestor DAG, capped at
1301/// [`MAX_ANCESTOR_DEPTH`] levels.
1302fn build_ancestor_chain(
1303    hashi: &Hashi,
1304    producing_id: Address,
1305    withdrawal_txns: &BTreeMap<Address, WithdrawalTransaction>,
1306    tx_confirmations: &HashMap<Address, u32>,
1307    utxo_records: &BTreeMap<UtxoId, UtxoRecord>,
1308) -> Vec<AncestorTx> {
1309    let mut chain = Vec::new();
1310    let mut queue = std::collections::VecDeque::new();
1311    queue.push_back((producing_id, 0usize));
1312
1313    while let Some((wid, depth)) = queue.pop_front() {
1314        if depth >= MAX_ANCESTOR_DEPTH {
1315            continue;
1316        }
1317
1318        let Some(txn) = withdrawal_txns.get(&wid) else {
1319            continue;
1320        };
1321
1322        let Ok(tx) = hashi.build_unsigned_withdrawal_tx(&txn.inputs, &txn.all_outputs()) else {
1323            continue;
1324        };
1325
1326        let confirmations = tx_confirmations.get(&wid).copied().unwrap_or(0);
1327        let input_total: u64 = txn.inputs.iter().map(|u| u.amount).sum();
1328        let output_total: u64 = txn.all_outputs().iter().map(|o| o.amount).sum();
1329
1330        chain.push(AncestorTx {
1331            confirmations,
1332            tx_weight: tx.weight(),
1333            tx_fee: input_total.saturating_sub(output_total),
1334        });
1335
1336        for input_utxo in &txn.inputs {
1337            if let Some(input_record) = utxo_records.get(&input_utxo.id)
1338                && let Some(parent_id) = input_record.produced_by
1339            {
1340                queue.push_back((parent_id, depth + 1));
1341            }
1342        }
1343    }
1344
1345    chain
1346}

1348/// Convert raw bitcoin address bytes (witness program) to a `ScriptBuf`.
1349/// 32-byte addresses are P2TR (witness v1), 20-byte addresses are P2WPKH (witness v0).
1350fn script_pubkey_from_raw_address(address_bytes: &[u8]) -> anyhow::Result<bitcoin::ScriptBuf> {
1351    let version = match address_bytes.len() {
1352        32 => WitnessVersion::V1,
1353        20 => WitnessVersion::V0,
1354        len => anyhow::bail!("Unsupported bitcoin address length: {len}"),
1355    };
1356    let program = WitnessProgram::new(version, address_bytes)
1357        .map_err(|e| anyhow!("Invalid witness program: {e}"))?;
1358    Ok(bitcoin::ScriptBuf::new_witness_program(&program))
1359}

1361/// Deterministic from on-chain state and the leader-supplied
1362/// `(timestamp_secs, seq)`, so every validator reconstructs the same request.
1363pub fn build_guardian_withdrawal_request(
1364    hashi: &Hashi,
1365    txn: &WithdrawalTransaction,
1366    timestamp_secs: u64,
1367    seq: u64,
1368) -> anyhow::Result<hashi_types::guardian::StandardWithdrawalRequest> {
1369    use hashi_types::guardian::bitcoin_utils::InputUTXO;
1370    use hashi_types::guardian::bitcoin_utils::OutputUTXO;
1371    use hashi_types::guardian::bitcoin_utils::TxUTXOs;
1372
1373    let hashi_pubkey = hashi.get_hashi_pubkey()?;
1374    let network = hashi.config.bitcoin_network();
1375
1376    let inputs = txn
1377        .inputs
1378        .iter()
1379        .map(|utxo| {
1380            let pubkey = hashi.deposit_pubkey(&hashi_pubkey, utxo.derivation_path.as_ref())?;
1381            let (_, _, leaf_hash) =
1382                bitcoin_utils::single_key_taproot_script_path_spend_artifacts(&pubkey);
1383            let address = hashi.bitcoin_address_from_pubkey(&pubkey);
1384
1385            let outpoint = bitcoin::OutPoint {
1386                txid: utxo.id.txid.into(),
1387                vout: utxo.id.vout,
1388            };
1389
1390            InputUTXO::new(
1391                outpoint,
1392                Amount::from_sat(utxo.amount),
1393                address.into_unchecked(),
1394                leaf_hash,
1395                network,
1396            )
1397            .map_err(|e| anyhow!("Failed to build guardian InputUTXO: {e}"))
1398        })
1399        .collect::<anyhow::Result<Vec<_>>>()?;
1400
1401    // First N outputs are external payouts; any trailing output is internal change.
1402    let all_outputs = txn.all_outputs();
1403    let num_requests = txn.request_ids.len();
1404    let outputs = all_outputs
1405        .iter()
1406        .enumerate()
1407        .map(|(i, output)| {
1408            if i < num_requests {
1409                let script_pubkey = script_pubkey_from_raw_address(&output.bitcoin_address)?;
1410                let address = BitcoinAddress::from_script(&script_pubkey, network)
1411                    .map_err(|e| anyhow!("Cannot derive address from output script: {e}"))?;
1412                OutputUTXO::new_external(
1413                    address.into_unchecked(),
1414                    Amount::from_sat(output.amount),
1415                    network,
1416                )
1417                .map_err(|e| anyhow!("Failed to build guardian external OutputUTXO: {e}"))
1418            } else {
1419                let derivation_path = [0u8; 32];
1420                Ok(OutputUTXO::new_internal(
1421                    derivation_path,
1422                    Amount::from_sat(output.amount),
1423                ))
1424            }
1425        })
1426        .collect::<anyhow::Result<Vec<_>>>()?;
1427
1428    let utxos = TxUTXOs::new(inputs, outputs)
1429        .map_err(|e| anyhow!("Failed to build guardian TxUTXOs: {e}"))?;
1430
1431    // The on-chain `WithdrawalTransaction` UID doubles as the guardian-side `wid`.
1432    let wid: hashi_types::guardian::WithdrawalID = txn.id;
1433
1434    Ok(hashi_types::guardian::StandardWithdrawalRequest::new(
1435        wid,
1436        utxos,
1437        timestamp_secs,
1438        seq,
1439    ))
1440}
