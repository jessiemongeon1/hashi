22/// Derive a deposit address from a compressed MPC public key (33-byte ProjectivePoint),
23/// an optional derivation path, and the target Bitcoin network.
24pub fn derive_deposit_address(
25    mpc_key: &ProjectivePoint,
26    derivation_path: Option<&sui_sdk_types::Address>,
27    btc_network: bitcoin::Network,
28) -> anyhow::Result<bitcoin::Address> {
29    let xonly = if let Some(path) = derivation_path {
30        let derived = fastcrypto_tbls::threshold_schnorr::key_derivation::derive_verifying_key(
31            mpc_key,
32            &path.into_inner(),
33        );
34        XOnlyPublicKey::from_slice(&derived.to_byte_array()).context("valid 32-byte x-only key")?
35    } else {
36        let schnorr_pk = SchnorrPublicKey::try_from(mpc_key)
37            .context("Failed to convert MPC key to schnorr key")?;
38        XOnlyPublicKey::from_slice(&schnorr_pk.to_byte_array())
39            .context("Failed to parse x-only key")?
40    };
41
42    Ok(bitcoin_utils::single_key_taproot_script_path_address(
43        &xonly,
44        btc_network,
45    ))
46}

58    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
59    pub async fn validate_deposit_request(
60        &self,
61        deposit_request: &DepositRequest,
62    ) -> Result<(), DepositError> {
63        self.validate_deposit_request_on_sui(deposit_request)?;
64        self.validate_deposit_request_on_bitcoin(deposit_request)
65            .await?;
66        self.screen_deposit(deposit_request).await?;
67        Ok(())
68    }

70    /// Run AML/Sanctions checks for the deposit request.
71    /// If no screener client is configured, checks are skipped.
72    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
73    async fn screen_deposit(&self, deposit_request: &DepositRequest) -> Result<(), DepositError> {
74        let Some(screener) = self.screener_client() else {
75            tracing::debug!("AML checks skipped: no screener configured");
76            return Ok(());
77        };
78
79        // bitcoin
80        let source_tx_hash = deposit_request.utxo.id.txid.to_string();
81        let bitcoin_chain_id = self.config.bitcoin_chain_id().to_string();
82
83        // sui
84        let destination_address = deposit_request.id.to_string();
85        let sui_chain_id = self.config.sui_chain_id().to_string();
86
87        let approved = screener
88            .approve_deposit(
89                &source_tx_hash,
90                &destination_address,
91                &bitcoin_chain_id,
92                &sui_chain_id,
93            )
94            .await
95            .map_err(|e| DepositError::AmlServiceError(anyhow!(e)))?;
96
97        if !approved {
98            return Err(DepositError::AmlRejected(anyhow!(
99                "AML checks failed for source tx {source_tx_hash}, destination {destination_address}, bitcoin chain {bitcoin_chain_id}, sui chain {sui_chain_id}"
100            )));
101        }
102
103        Ok(())
104    }

106    /// Validate that the deposit request exists on Sui
107    #[tracing::instrument(level = "debug", skip_all, fields(deposit_id = %deposit_request.id))]
108    fn validate_deposit_request_on_sui(
109        &self,
110        deposit_request: &DepositRequest,
111    ) -> Result<(), DepositError> {
112        let state = self.onchain_state().state();
113        let deposit_queue = &state.hashi().deposit_queue;
114        match deposit_queue.requests().get(&deposit_request.id) {
115            None => {
116                return Err(DepositError::InvalidOnchainRequest(anyhow!(
117                    "Deposit request not found on Sui"
118                )));
119            }
120            Some(onchain_request) => {
121                // Approval state lives on the on-chain request but is not
122                // carried by `DepositRequest` constructed from the proto, so
123                // compare the immutable core fields only.
124                let core_matches = onchain_request.id == deposit_request.id
125                    && onchain_request.sender == deposit_request.sender
126                    && onchain_request.timestamp_ms == deposit_request.timestamp_ms
127                    && onchain_request.sui_tx_digest == deposit_request.sui_tx_digest
128                    && onchain_request.utxo == deposit_request.utxo;
129                if !core_matches {
130                    return Err(DepositError::InvalidOnchainRequest(anyhow!(
131                        "Deposit request fields do not match on-chain state"
132                    )));
133                }
134
135                // Refuse to sign a re-approval if the on-chain request is
136                // already approved by the current committee. The on-chain
137                // `approve_deposit` would reject it anyway, so we don't
138                // want to waste a signature exchange or a transaction.
139                let current_epoch = self.onchain_state().epoch();
140                if let Some(cert) = &onchain_request.approval_cert
141                    && cert.epoch == current_epoch
142                {
143                    return Err(DepositError::AlreadyApprovedThisEpoch);
144                }
145            }
146        }
147
148        let utxo_pool = &state.hashi().utxo_pool;
149        if utxo_pool
150            .utxo_records()
151            .contains_key(&deposit_request.utxo.id)
152            || utxo_pool
153                .spent_utxos()
154                .contains_key(&deposit_request.utxo.id)
155        {
156            return Err(DepositError::DuplicateOrSpentOnSui(anyhow!(
157                "UTXO {:?} is already active or spent",
158                deposit_request.utxo.id
159            )));
160        }
161
162        Ok(())
163    }
164
165    /// Validate that there is a txout on Bitcoin that matches the deposit request
166    #[tracing::instrument(
167        level = "debug",
168        skip_all,
169        fields(
170            deposit_id = %deposit_request.id,
171            bitcoin_txid = %deposit_request.utxo.id.txid,
172            vout = deposit_request.utxo.id.vout,
173        ),
174    )]
175    async fn validate_deposit_request_on_bitcoin(
176        &self,
177        deposit_request: &DepositRequest,
178    ) -> Result<(), DepositError> {
179        let outpoint = bitcoin::OutPoint {
180            txid: deposit_request.utxo.id.txid.into(),
181            vout: deposit_request.utxo.id.vout,
182        };
183        // Read the threshold live from on-chain state so governance
184        // updates take effect for new deposits without a restart.
185        let confirmation_threshold = self.onchain_state().bitcoin_confirmation_threshold();
186        let txout = self
187            .btc_monitor()
188            .confirm_deposit(outpoint, confirmation_threshold)
189            .await
190            .map_err(|e| match e {
191                DepositConfirmError::UtxoSpent { .. } => {
192                    self.metrics.deposits_rejected_utxo_spent.inc();
193                    DepositError::BitcoinUtxoSpent(anyhow!(e))
194                }
195                DepositConfirmError::Other(err) => DepositError::BitcoinConfirmFailed(err),
196            })?;
197        if txout.value.to_sat() != deposit_request.utxo.amount {
198            return Err(DepositError::DepositDataMismatch(anyhow!(
199                "Bitcoin deposit amount mismatch: got {}, onchain is {}",
200                deposit_request.utxo.amount,
201                txout.value.to_sat(),
202            )));
203        }
204
205        self.validate_deposit_request_derivation_path(&txout.script_pubkey, deposit_request)
206            .await?;
207        Ok(())
208    }

210    async fn validate_deposit_request_derivation_path(
211        &self,
212        script_pubkey: &ScriptBuf,
213        deposit_request: &DepositRequest,
214    ) -> Result<(), DepositError> {
215        let deposit_address =
216            bitcoin::Address::from_script(script_pubkey, self.config.bitcoin_network()).map_err(
217                |e| {
218                    DepositError::DepositDataMismatch(anyhow!(
219                        "Failed to extract address from script_pubkey: {e}"
220                    ))
221                },
222            )?;
223        let hashi_pubkey = self.get_hashi_pubkey().map_err(DepositError::NotReady)?;
224        let expected_address = self
225            .get_deposit_address(&hashi_pubkey, deposit_request.utxo.derivation_path.as_ref())
226            .map_err(DepositError::DepositDataMismatch)?;
227
228        if deposit_address != expected_address {
229            return Err(DepositError::DepositDataMismatch(anyhow!(
230                "Expected address {expected_address}, got address {deposit_address}",
231            )));
232        }
233
234        Ok(())
235    }

237    pub fn get_deposit_address(
238        &self,
239        hashi_pubkey: &XOnlyPublicKey,
240        derivation_path: Option<&sui_sdk_types::Address>,
241    ) -> anyhow::Result<bitcoin::Address> {
242        let pubkey = self.deposit_pubkey(hashi_pubkey, derivation_path)?;
243        Ok(self.bitcoin_address_from_pubkey(&pubkey))
244    }

246    pub(crate) fn deposit_pubkey(
247        &self,
248        hashi_pubkey: &XOnlyPublicKey,
249        derivation_path: Option<&sui_sdk_types::Address>,
250    ) -> anyhow::Result<XOnlyPublicKey> {
251        if let Some(path) = derivation_path {
252            // Prefer the local signing manager (available after DKG preparation).
253            // Fall back to the on-chain key, which is guaranteed present once the
254            // initial committee has formed and `end_reconfig` has been processed.
255            let verifying_key = self
256                .signing_verifying_key()
257                .map(Ok)
258                .unwrap_or_else(|| self.onchain_verifying_key_g())
259                .context("MPC public key not available yet")?;
260            let derived = fastcrypto_tbls::threshold_schnorr::key_derivation::derive_verifying_key(
261                &verifying_key,
262                &path.into_inner(),
263            );
264            let pubkey = XOnlyPublicKey::from_slice(&derived.to_byte_array())
265                .context("valid 32-byte x-only key")?;
266            Ok(pubkey)
267        } else {
268            Ok(*hashi_pubkey)
269        }
270    }

276    pub fn get_hashi_pubkey(&self) -> anyhow::Result<XOnlyPublicKey> {
277        let g = self
278            .mpc_handle()
279            .context("MpcHandle not initialized")?
280            .public_key()
281            .context("MPC public key not available yet")?;
282        // Convert G (ProjectivePoint, 33 bytes compressed) to SchnorrPublicKey (32 bytes x-only)
283        let schnorr_pk = SchnorrPublicKey::try_from(&g)
284            .map_err(|e| anyhow!("invalid group element for schnorr key: {e}"))?;
285        Ok(XOnlyPublicKey::from_slice(&schnorr_pk.to_byte_array())?)
286    }

288    /// Read the MPC public key from on-chain state.
289    ///
290    /// Unlike `get_hashi_pubkey`, this does not depend on the node's local DKG
291    /// completion channel. The on-chain key is set during `end_reconfig` event
292    /// processing, so it is guaranteed to be present once the initial committee
293    /// has formed (i.e., after `HashiNetworkBuilder::build()` returns).
294    pub fn get_onchain_mpc_pubkey(&self) -> anyhow::Result<XOnlyPublicKey> {
295        let g = self.onchain_verifying_key_g()?;
296        let schnorr_pk = SchnorrPublicKey::try_from(&g)
297            .map_err(|e| anyhow!("invalid group element for schnorr key: {e}"))?;
298        Ok(XOnlyPublicKey::from_slice(&schnorr_pk.to_byte_array())?)
299    }

301    /// Deserialize the BCS-encoded MPC group element from on-chain state.
302    ///
303    /// The on-chain key is stored as `bcs::to_bytes(&G)` in the `CommitteeSet`
304    /// and is populated atomically with the `end_reconfig` event.
305    fn onchain_verifying_key_g(&self) -> anyhow::Result<G> {
306        let bytes = self.onchain_state().mpc_public_key();
307        anyhow::ensure!(
308            !bytes.is_empty(),
309            "MPC public key not yet available on-chain"
310        );
311        bcs::from_bytes(&bytes).context("failed to deserialize on-chain MPC public key")
312    }

314    fn sign_deposit_confirmation(
315        &self,
316        deposit_request: &DepositRequest,
317    ) -> anyhow::Result<MemberSignature> {
318        let epoch = self.onchain_state().epoch();
319        let validator_address = self
320            .config
321            .validator_address()
322            .map_err(|e| anyhow!("No validator address configured: {}", e))?;
323        let committee = self
324            .onchain_state()
325            .state()
326            .hashi()
327            .committees
328            .committees()
329            .get(&epoch)
330            .cloned()
331            .ok_or_else(|| anyhow!("no committee for epoch {epoch}"))?;
332        let private_key =
333            self.find_signing_key_for_committee(&committee, validator_address, epoch)?;
334        let public_key_bytes = private_key.public_key().as_bytes().to_vec().into();
335
336        let message = DepositConfirmationMessage {
337            request_id: deposit_request.id,
338            utxo: deposit_request.utxo.clone(),
339        };
340
341        let signature_bytes = private_key
342            .sign(epoch, validator_address, &message)
343            .signature()
344            .as_bytes()
345            .to_vec()
346            .into();
347
348        Ok(MemberSignature {
349            epoch: Some(epoch),
350            address: Some(validator_address.to_string()),
351            public_key: Some(public_key_bytes),
352            signature: Some(signature_bytes),
353        })
354    }

357#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
358pub(crate) enum DepositErrorKind {
359    RetryOnNextBlock,
360    NeverRetry,
361}
362
363#[derive(Debug, Error)]
364pub enum DepositError {
365    #[error("Failed to confirm Bitcoin deposit: {0}")]
366    BitcoinConfirmFailed(#[source] anyhow::Error),
367
368    #[error("Screener service error: {0}")]
369    AmlServiceError(#[source] anyhow::Error),
370
371    #[error("Not ready: {0}")]
372    NotReady(#[source] anyhow::Error),
373
374    #[error("Invalid on-chain deposit request: {0}")]
375    InvalidOnchainRequest(#[source] anyhow::Error),
376
377    #[error("UTXO is already active or spent on Sui: {0}")]
378    DuplicateOrSpentOnSui(#[source] anyhow::Error),
379
380    #[error("Deposit UTXO has already been spent on Bitcoin: {0}")]
381    BitcoinUtxoSpent(#[source] anyhow::Error),
382
383    #[error("Deposit data mismatch: {0}")]
384    DepositDataMismatch(#[source] anyhow::Error),
385
386    #[error("AML checks rejected deposit: {0}")]
387    AmlRejected(#[source] anyhow::Error),
388
389    #[error("Failed quorum: weight {weight} < {required_weight}")]
390    FailedQuorum { weight: u64, required_weight: u64 },
391
392    #[error("Failed to build deposit certificate: {0}")]
393    CertificateBuildFailed(#[source] anyhow::Error),
394
395    #[error("Failed to create Sui transaction executor: {0}")]
396    ExecutorInitFailed(#[source] anyhow::Error),
397
398    #[error("Failed to approve deposit on Sui: {0}")]
399    ApproveDepositFailed(#[source] anyhow::Error),
400
401    #[error("Failed to confirm deposit on Sui: {0}")]
402    ConfirmDepositFailed(#[source] anyhow::Error),
403
404    #[error("Deposit has already been approved by the current committee")]
405    AlreadyApprovedThisEpoch,
406
407    #[error("Deposit processing timed out after {0:?}")]
408    TimedOut(std::time::Duration),
409}
410
411impl DepositError {
412    pub(crate) fn kind(&self) -> DepositErrorKind {
413        match self {
414            Self::BitcoinConfirmFailed(_)
415            | Self::AmlServiceError(_)
416            | Self::NotReady(_)
417            | Self::FailedQuorum { .. }
418            | Self::CertificateBuildFailed(_)
419            | Self::ExecutorInitFailed(_)
420            | Self::ApproveDepositFailed(_)
421            | Self::ConfirmDepositFailed(_)
422            | Self::AlreadyApprovedThisEpoch
423            | Self::TimedOut(_) => DepositErrorKind::RetryOnNextBlock,
424            Self::InvalidOnchainRequest(_)
425            | Self::DuplicateOrSpentOnSui(_)
426            | Self::BitcoinUtxoSpent(_)
427            | Self::DepositDataMismatch(_)
428            | Self::AmlRejected(_) => DepositErrorKind::NeverRetry,
429        }
430    }
